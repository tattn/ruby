179  * call-seq:
180  * RubyVM.stat -> Hash
181  * RubyVM.stat(hsh) -> hsh
182  * RubyVM.stat(Symbol) -> Numeric
183  *
184  * Returns a Hash containing implementation-dependent counters inside the VM.
185  *
186  * This hash includes information about method/constant cache serials:
187  *
188  * {
189  * :global_method_state=>251,
190  * :global_constant_state=>481,
191  * :class_serial=>9029
192  * }
193  *
194  * The contents of the hash are implementation specific and may be changed in
195  * the future.
196  *
197  * This method is only expected to work on C Ruby.
198  */

200 static VALUE
201 vm_stat(int argc, VALUE *argv, VALUE self)
202 {
203  static VALUE sym_global_method_state, sym_global_constant_state, sym_class_serial;
204  VALUE arg = Qnil;
205  VALUE hash = Qnil, key = Qnil;
206 
207  if (rb_scan_args(argc, argv, "01", &arg) == 1) {
208  if (SYMBOL_P(arg))
209  key = arg;
210  else if (RB_TYPE_P(arg, T_HASH))
211  hash = arg;
212  else
213  rb_raise(rb_eTypeError, "non-hash or symbol given");
214  }
215  else if (NIL_P(arg)) {
216  hash = rb_hash_new();
217  }
218 
219  if (sym_global_method_state == 0) {
220 #define S(s) sym_##s = ID2SYM(rb_intern_const(#s))
221  S(global_method_state);
222  S(global_constant_state);
223  S(class_serial);
224 #undef S
225  }
226 
227 #define SET(name, attr) \
228  if (key == sym_##name) \
229  return SERIALT2NUM(attr); \
230  else if (hash != Qnil) \
231  rb_hash_aset(hash, sym_##name, SERIALT2NUM(attr));
232 
233  SET(global_method_state, ruby_vm_global_method_state);
234  SET(global_constant_state, ruby_vm_global_constant_state);
235  SET(class_serial, ruby_vm_class_serial);
236 #undef SET
237 
238  if (!NIL_P(key)) { /* matched key should return above */
239  rb_raise(rb_eArgError, "unknown key: %"PRIsVALUE, rb_sym2str(key));
240  }
241 
242  return hash;
243 }

810 static inline VALUE
811 invoke_block_from_c(rb_thread_t *th, const rb_block_t *block,
812  VALUE self, int argc, const VALUE *argv,
813  const rb_block_t *blockptr, const rb_cref_t *cref,
814  VALUE defined_class, int splattable)
815 {
816  if (SPECIAL_CONST_P(block->iseq)) {
817  return Qnil;
818  }
819  else if (!RUBY_VM_IFUNC_P(block->iseq)) {
820  VALUE ret;
821  const rb_iseq_t *iseq = block->iseq;
822  const rb_control_frame_t *cfp;
823  int i, opt_pc, arg_size = iseq->param.size;
824  int type = block_proc_is_lambda(block->proc) ? VM_FRAME_MAGIC_LAMBDA : VM_FRAME_MAGIC_BLOCK;
825  const rb_method_entry_t *me = th->passed_bmethod_me;
826  th->passed_bmethod_me = NULL;
827  cfp = th->cfp;
828 
829  for (i=0; i<argc; i++) {
830  cfp->sp[i] = argv[i];
831  }
832 
833  opt_pc = vm_yield_setup_args(th, iseq, argc, cfp->sp, blockptr,
834  (type == VM_FRAME_MAGIC_LAMBDA ? (splattable ? arg_setup_lambda : arg_setup_method) : arg_setup_block));
835 
836  if (me != 0) {
837  /* bmethod */
838  vm_push_frame(th, iseq, type | VM_FRAME_FLAG_FINISH | VM_FRAME_FLAG_BMETHOD,
839  self, defined_class,
840  VM_ENVVAL_PREV_EP_PTR(block->ep),
841  (VALUE)me, /* cref or method (TODO: can we ignore cref?) */
842  iseq->iseq_encoded + opt_pc,
843  cfp->sp + arg_size, iseq->local_size - arg_size,
844  iseq->stack_max);
845 
846  RUBY_DTRACE_METHOD_ENTRY_HOOK(th, me->klass, me->called_id);
847  EXEC_EVENT_HOOK(th, RUBY_EVENT_CALL, self, me->called_id, me->klass, Qnil);
848  }
849  else {
850  vm_push_frame(th, iseq, type | VM_FRAME_FLAG_FINISH,
851  self, defined_class,
852  VM_ENVVAL_PREV_EP_PTR(block->ep),
853  (VALUE)cref, /* cref or method */
854  iseq->iseq_encoded + opt_pc,
855  cfp->sp + arg_size, iseq->local_size - arg_size,
856  iseq->stack_max);
857  }
858 
859  ret = vm_exec(th);
860 
861  if (me) {
862  /* bmethod */
863  EXEC_EVENT_HOOK(th, RUBY_EVENT_RETURN, self, me->called_id, me->klass, ret);
864  RUBY_DTRACE_METHOD_RETURN_HOOK(th, me->klass, me->called_id);
865  }
866 
867  return ret;
868  }
869  else {
870  return vm_yield_with_cfunc(th, block, self, defined_class,
871  argc, argv, blockptr);
872  }
873 }

1473 static VALUE
1474 vm_exec(rb_thread_t *th)
1475 {
1476  int state;
1477  VALUE result;
1478  VALUE initial = 0;
1479  struct vm_throw_data *err;
1480 
1481  TH_PUSH_TAG(th);
1482  _tag.retval = Qnil;
1483  /* setjmp でエラー時に戻れるようにしている。state には th->state が入る。 */
1484  if ((state = EXEC_TAG()) == 0) {
1485  vm_loop_start:
1486  result = vm_exec_core(th, initial);
1487  if ((state = th->state) != 0) {
1488  err = (struct vm_throw_data *)result;
1489  th->state = 0;
1490  goto exception_handler;
1491  }
1492  }
1493  else {
1494  int i;
1495  struct iseq_catch_table_entry *entry;
1496  struct iseq_catch_table *ct;
1497  unsigned long epc, cont_pc, cont_sp;
1498  VALUE catch_iseqval;
1499  rb_control_frame_t *cfp;
1500  VALUE type;
1501  const rb_control_frame_t *escape_cfp;
1502 
1503  err = (struct vm_throw_data *)th->errinfo;
1504 
1505  exception_handler:
1506  cont_pc = cont_sp = catch_iseqval = 0;
1507 
1508  while (th->cfp->pc == 0 || th->cfp->iseq == 0) {
1509  if (UNLIKELY(VM_FRAME_TYPE(th->cfp) == VM_FRAME_MAGIC_CFUNC)) {
1510  EXEC_EVENT_HOOK(th, RUBY_EVENT_C_RETURN, th->cfp->self,
1511  rb_vm_frame_method_entry(th->cfp)->called_id, rb_vm_frame_method_entry(th->cfp)->klass, Qnil);
1512  RUBY_DTRACE_METHOD_RETURN_HOOK(th, rb_vm_frame_method_entry(th->cfp)->klass,
1513  rb_vm_frame_method_entry(th->cfp)->called_id);
1514  }
1515  th->cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(th->cfp);
1516  }
1517 
1518  cfp = th->cfp;
1519  epc = cfp->pc - cfp->iseq->iseq_encoded;
1520 
1521  escape_cfp = NULL;
1522  if (state == TAG_BREAK || state == TAG_RETURN) {
1523  escape_cfp = THROW_DATA_CATCH_FRAME(err);
1524 
1525  if (cfp == escape_cfp) {
1526  if (state == TAG_RETURN) {
1527  if (!VM_FRAME_TYPE_FINISH_P(cfp)) {
1528  THROW_DATA_CATCH_FRAME_SET(err, cfp + 1);
1529  THROW_DATA_STATE_SET(err, state = TAG_BREAK);
1530  }
1531  else {
1532  ct = cfp->iseq->catch_table;
1533  if (ct) for (i = 0; i < ct->size; i++) {
1534  entry = &ct->entries[i];
1535  if (entry->start < epc && entry->end >= epc) {
1536  if (entry->type == CATCH_TYPE_ENSURE) {
1537  catch_iseqval = entry->iseq;
1538  cont_pc = entry->cont;
1539  cont_sp = entry->sp;
1540  break;
1541  }
1542  }
1543  }
1544  if (!catch_iseqval) {
1545  th->errinfo = Qnil;
1546  result = THROW_DATA_VAL(err);
1547  hook_before_rewind(th, th->cfp);
1548  vm_pop_frame(th);
1549  goto finish_vme;
1550  }
1551  }
1552  /* through */
1553  }
1554  else {
1555  /* TAG_BREAK */
1556 #if OPT_STACK_CACHING
1557  initial = THROW_DATA_VAL(err);
1558 #else
1559  *th->cfp->sp++ = THROW_DATA_VAL(err);
1560 #endif
1561  th->errinfo = Qnil;
1562  goto vm_loop_start;
1563  }
1564  }
1565  }
1566 
1567  if (state == TAG_RAISE) {
1568  ct = cfp->iseq->catch_table;
1569  if (ct) for (i = 0; i < ct->size; i++) {
1570  entry = &ct->entries[i];
1571  if (entry->start < epc && entry->end >= epc) {
1572 
1573  if (entry->type == CATCH_TYPE_RESCUE ||
1574  entry->type == CATCH_TYPE_ENSURE) {
1575  catch_iseqval = entry->iseq;
1576  cont_pc = entry->cont;
1577  cont_sp = entry->sp;
1578  break;
1579  }
1580  }
1581  }
1582  }
1583  else if (state == TAG_RETRY) {
1584  ct = cfp->iseq->catch_table;
1585  if (ct) for (i = 0; i < ct->size; i++) {
1586  entry = &ct->entries[i];
1587  if (entry->start < epc && entry->end >= epc) {
1588 
1589  if (entry->type == CATCH_TYPE_ENSURE) {
1590  catch_iseqval = entry->iseq;
1591  cont_pc = entry->cont;
1592  cont_sp = entry->sp;
1593  break;
1594  }
1595  else if (entry->type == CATCH_TYPE_RETRY) {
1596  const rb_control_frame_t *escape_cfp;
1597  escape_cfp = THROW_DATA_CATCH_FRAME(err);
1598  if (cfp == escape_cfp) {
1599  cfp->pc = cfp->iseq->iseq_encoded + entry->cont;
1600  th->errinfo = Qnil;
1601  goto vm_loop_start;
1602  }
1603  }
1604  }
1605  }
1606  }
1607  else if (state == TAG_BREAK && !escape_cfp) {
1608  type = CATCH_TYPE_BREAK;
1609 
1610  search_restart_point:
1611  ct = cfp->iseq->catch_table;
1612  if (ct) for (i = 0; i < ct->size; i++) {
1613  entry = &ct->entries[i];
1614 
1615  if (entry->start < epc && entry->end >= epc) {
1616  if (entry->type == CATCH_TYPE_ENSURE) {
1617  catch_iseqval = entry->iseq;
1618  cont_pc = entry->cont;
1619  cont_sp = entry->sp;
1620  break;
1621  }
1622  else if (entry->type == type) {
1623  cfp->pc = cfp->iseq->iseq_encoded + entry->cont;
1624  cfp->sp = vm_base_ptr(cfp) + entry->sp;
1625 
1626  if (state != TAG_REDO) {
1627 #if OPT_STACK_CACHING
1628  initial = THROW_DATA_VAL(err);
1629 #else
1630  *th->cfp->sp++ = THROW_DATA_VAL(err);
1631 #endif
1632  }
1633  th->errinfo = Qnil;
1634  th->state = 0;
1635  goto vm_loop_start;
1636  }
1637  }
1638  }
1639  }
1640  else if (state == TAG_REDO) {
1641  type = CATCH_TYPE_REDO;
1642  goto search_restart_point;
1643  }
1644  else if (state == TAG_NEXT) {
1645  type = CATCH_TYPE_NEXT;
1646  goto search_restart_point;
1647  }
1648  else {
1649  ct = cfp->iseq->catch_table;
1650  if (ct) for (i = 0; i < ct->size; i++) {
1651  entry = &ct->entries[i];
1652  if (entry->start < epc && entry->end >= epc) {
1653 
1654  if (entry->type == CATCH_TYPE_ENSURE) {
1655  catch_iseqval = entry->iseq;
1656  cont_pc = entry->cont;
1657  cont_sp = entry->sp;
1658  break;
1659  }
1660  }
1661  }
1662  }
1663 
1664  if (catch_iseqval != 0) {
1665  /* found catch table */
1666  rb_iseq_t *catch_iseq;
1667 
1668  /* enter catch scope */
1669  GetISeqPtr(catch_iseqval, catch_iseq);
1670  cfp->sp = vm_base_ptr(cfp) + cont_sp;
1671  cfp->pc = cfp->iseq->iseq_encoded + cont_pc;
1672 
1673  /* push block frame */
1674  cfp->sp[0] = (VALUE)err;
1675  vm_push_frame(th, catch_iseq, VM_FRAME_MAGIC_RESCUE,
1676  cfp->self, cfp->klass,
1677  VM_ENVVAL_PREV_EP_PTR(cfp->ep),
1678  0, /* cref or me */
1679  catch_iseq->iseq_encoded,
1680  cfp->sp + 1 /* push value */,
1681  catch_iseq->local_size - 1,
1682  catch_iseq->stack_max);
1683 
1684  state = 0;
1685  th->state = 0;
1686  th->errinfo = Qnil;
1687  goto vm_loop_start;
1688  }
1689  else {
1690  /* skip frame */
1691  hook_before_rewind(th, th->cfp);
1692 
1693  if (VM_FRAME_TYPE_FINISH_P(th->cfp)) {
1694  vm_pop_frame(th);
1695  th->errinfo = (VALUE)err;
1696  TH_TMPPOP_TAG();
1697  JUMP_TAG(state);
1698  }
1699  else {
1700  th->cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(th->cfp);
1701  goto exception_handler;
1702  }
1703  }
1704  }
1705  finish_vme:
1706  TH_POP_TAG();
1707  return result;
1708 }

2569  /*
2570  * Document-class: Thread
2571  *
2572  * Threads are the Ruby implementation for a concurrent programming model.
2573  *
2574  * Programs that require multiple threads of execution are a perfect
2575  * candidate for Ruby's Thread class.
2576  *
2577  * For example, we can create a new thread separate from the main thread's
2578  * execution using ::new.
2579  *
2580  * thr = Thread.new { puts "Whats the big deal" }
2581  *
2582  * Then we are able to pause the execution of the main thread and allow
2583  * our new thread to finish, using #join:
2584  *
2585  * thr.join #=> "Whats the big deal"
2586  *
2587  * If we don't call +thr.join+ before the main thread terminates, then all
2588  * other threads including +thr+ will be killed.
2589  *
2590  * Alternatively, you can use an array for handling multiple threads at
2591  * once, like in the following example:
2592  *
2593  * threads = []
2594  * threads << Thread.new { puts "Whats the big deal" }
2595  * threads << Thread.new { 3.times { puts "Threads are fun!" } }
2596  *
2597  * After creating a few threads we wait for them all to finish
2598  * consecutively.
2599  *
2600  * threads.each { |thr| thr.join }
2601  *
2602  * === Thread initialization
2603  *
2604  * In order to create new threads, Ruby provides ::new, ::start, and
2605  * ::fork. A block must be provided with each of these methods, otherwise
2606  * a ThreadError will be raised.
2607  *
2608  * When subclassing the Thread class, the +initialize+ method of your
2609  * subclass will be ignored by ::start and ::fork. Otherwise, be sure to
2610  * call super in your +initialize+ method.
2611  *
2612  * === Thread termination
2613  *
2614  * For terminating threads, Ruby provides a variety of ways to do this.
2615  *
2616  * The class method ::kill, is meant to exit a given thread:
2617  *
2618  * thr = Thread.new { ... }
2619  * Thread.kill(thr) # sends exit() to thr
2620  *
2621  * Alternatively, you can use the instance method #exit, or any of its
2622  * aliases #kill or #terminate.
2623  *
2624  * thr.exit
2625  *
2626  * === Thread status
2627  *
2628  * Ruby provides a few instance methods for querying the state of a given
2629  * thread. To get a string with the current thread's state use #status
2630  *
2631  * thr = Thread.new { sleep }
2632  * thr.status # => "sleep"
2633  * thr.exit
2634  * thr.status # => false
2635  *
2636  * You can also use #alive? to tell if the thread is running or sleeping,
2637  * and #stop? if the thread is dead or sleeping.
2638  *
2639  * === Thread variables and scope
2640  *
2641  * Since threads are created with blocks, the same rules apply to other
2642  * Ruby blocks for variable scope. Any local variables created within this
2643  * block are accessible to only this thread.
2644  *
2645  * ==== Fiber-local vs. Thread-local
2646  *
2647  * Each fiber has its own bucket for Thread#[] storage. When you set a
2648  * new fiber-local it is only accessible within this Fiber. To illustrate:
2649  *
2650  * Thread.new {
2651  * Thread.current[:foo] = "bar"
2652  * Fiber.new {
2653  * p Thread.current[:foo] # => nil
2654  * }.resume
2655  * }.join
2656  *
2657  * This example uses #[] for getting and #[]= for setting fiber-locals,
2658  * you can also use #keys to list the fiber-locals for a given
2659  * thread and #key? to check if a fiber-local exists.
2660  *
2661  * When it comes to thread-locals, they are accessible within the entire
2662  * scope of the thread. Given the following example:
2663  *
2664  * Thread.new{
2665  * Thread.current.thread_variable_set(:foo, 1)
2666  * p Thread.current.thread_variable_get(:foo) # => 1
2667  * Fiber.new{
2668  * Thread.current.thread_variable_set(:foo, 2)
2669  * p Thread.current.thread_variable_get(:foo) # => 2
2670  * }.resume
2671  * p Thread.current.thread_variable_get(:foo) # => 2
2672  * }.join
2673  *
2674  * You can see that the thread-local +:foo+ carried over into the fiber
2675  * and was changed to +2+ by the end of the thread.
2676  *
2677  * This example makes use of #thread_variable_set to create new
2678  * thread-locals, and #thread_variable_get to reference them.
2679  *
2680  * There is also #thread_variables to list all thread-locals, and
2681  * #thread_variable? to check if a given thread-local exists.
2682  *
2683  * === Exception handling
2684  *
2685  * Any thread can raise an exception using the #raise instance method,
2686  * which operates similarly to Kernel#raise.
2687  *
2688  * However, it's important to note that an exception that occurs in any
2689  * thread except the main thread depends on #abort_on_exception. This
2690  * option is +false+ by default, meaning that any unhandled exception will
2691  * cause the thread to terminate silently when waited on by either #join
2692  * or #value. You can change this default by either #abort_on_exception=
2693  * +true+ or setting $DEBUG to +true+.
2694  *
2695  * With the addition of the class method ::handle_interrupt, you can now
2696  * handle exceptions asynchronously with threads.
2697  *
2698  * === Scheduling
2699  *
2700  * Ruby provides a few ways to support scheduling threads in your program.
2701  *
2702  * The first way is by using the class method ::stop, to put the current
2703  * running thread to sleep and schedule the execution of another thread.
2704  *
2705  * Once a thread is asleep, you can use the instance method #wakeup to
2706  * mark your thread as eligible for scheduling.
2707  *
2708  * You can also try ::pass, which attempts to pass execution to another
2709  * thread but is dependent on the OS whether a running thread will switch
2710  * or not. The same goes for #priority, which lets you hint to the thread
2711  * scheduler which threads you want to take precedence when passing
2712  * execution. This method is also dependent on the OS and may be ignored
2713  * on some platforms.
2714  *
2715  */
2716  rb_cThread = rb_define_class("Thread", rb_cObject);
