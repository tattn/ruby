185 /*
186  * call-seq:
187  * format(format_string [, arguments...] ) -> string
188  * sprintf(format_string [, arguments...] ) -> string
189  *
190  * Returns the string resulting from applying <i>format_string</i> to
191  * any additional arguments. Within the format string, any characters
192  * other than format sequences are copied to the result.
193  *
194  * The syntax of a format sequence is follows.
195  *
196  * %[flags][width][.precision]type
197  *
198  * A format
199  * sequence consists of a percent sign, followed by optional flags,
200  * width, and precision indicators, then terminated with a field type
201  * character. The field type controls how the corresponding
202  * <code>sprintf</code> argument is to be interpreted, while the flags
203  * modify that interpretation.
204  *
205  * The field type characters are:
206  *
207  * Field | Integer Format
208  * ------+--------------------------------------------------------------
209  * b | Convert argument as a binary number.
210  * | Negative numbers will be displayed as a two's complement
211  * | prefixed with `..1'.
212  * B | Equivalent to `b', but uses an uppercase 0B for prefix
213  * | in the alternative format by #.
214  * d | Convert argument as a decimal number.
215  * i | Identical to `d'.
216  * o | Convert argument as an octal number.
217  * | Negative numbers will be displayed as a two's complement
218  * | prefixed with `..7'.
219  * u | Identical to `d'.
220  * x | Convert argument as a hexadecimal number.
221  * | Negative numbers will be displayed as a two's complement
222  * | prefixed with `..f' (representing an infinite string of
223  * | leading 'ff's).
224  * X | Equivalent to `x', but uses uppercase letters.
225  *
226  * Field | Float Format
227  * ------+--------------------------------------------------------------
228  * e | Convert floating point argument into exponential notation
229  * | with one digit before the decimal point as [-]d.dddddde[+-]dd.
230  * | The precision specifies the number of digits after the decimal
231  * | point (defaulting to six).
232  * E | Equivalent to `e', but uses an uppercase E to indicate
233  * | the exponent.
234  * f | Convert floating point argument as [-]ddd.dddddd,
235  * | where the precision specifies the number of digits after
236  * | the decimal point.
237  * g | Convert a floating point number using exponential form
238  * | if the exponent is less than -4 or greater than or
239  * | equal to the precision, or in dd.dddd form otherwise.
240  * | The precision specifies the number of significant digits.
241  * G | Equivalent to `g', but use an uppercase `E' in exponent form.
242  * a | Convert floating point argument as [-]0xh.hhhhp[+-]dd,
243  * | which is consisted from optional sign, "0x", fraction part
244  * | as hexadecimal, "p", and exponential part as decimal.
245  * A | Equivalent to `a', but use uppercase `X' and `P'.
246  *
247  * Field | Other Format
248  * ------+--------------------------------------------------------------
249  * c | Argument is the numeric code for a single character or
250  * | a single character string itself.
251  * p | The valuing of argument.inspect.
252  * s | Argument is a string to be substituted. If the format
253  * | sequence contains a precision, at most that many characters
254  * | will be copied.
255  * % | A percent sign itself will be displayed. No argument taken.
256  *
257  * The flags modifies the behavior of the formats.
258  * The flag characters are:
259  *
260  * Flag | Applies to | Meaning
261  * ---------+---------------+-----------------------------------------
262  * space | bBdiouxX | Leave a space at the start of
263  * | aAeEfgG | non-negative numbers.
264  * | (numeric fmt) | For `o', `x', `X', `b' and `B', use
265  * | | a minus sign with absolute value for
266  * | | negative values.
267  * ---------+---------------+-----------------------------------------
268  * (digit)$ | all | Specifies the absolute argument number
269  * | | for this field. Absolute and relative
270  * | | argument numbers cannot be mixed in a
271  * | | sprintf string.
272  * ---------+---------------+-----------------------------------------
273  * # | bBoxX | Use an alternative format.
274  * | aAeEfgG | For the conversions `o', increase the precision
275  * | | until the first digit will be `0' if
276  * | | it is not formatted as complements.
277  * | | For the conversions `x', `X', `b' and `B'
278  * | | on non-zero, prefix the result with ``0x'',
279  * | | ``0X'', ``0b'' and ``0B'', respectively.
280  * | | For `a', `A', `e', `E', `f', `g', and 'G',
281  * | | force a decimal point to be added,
282  * | | even if no digits follow.
283  * | | For `g' and 'G', do not remove trailing zeros.
284  * ---------+---------------+-----------------------------------------
285  * + | bBdiouxX | Add a leading plus sign to non-negative
286  * | aAeEfgG | numbers.
287  * | (numeric fmt) | For `o', `x', `X', `b' and `B', use
288  * | | a minus sign with absolute value for
289  * | | negative values.
290  * ---------+---------------+-----------------------------------------
291  * - | all | Left-justify the result of this conversion.
292  * ---------+---------------+-----------------------------------------
293  * 0 (zero) | bBdiouxX | Pad with zeros, not spaces.
294  * | aAeEfgG | For `o', `x', `X', `b' and `B', radix-1
295  * | (numeric fmt) | is used for negative numbers formatted as
296  * | | complements.
297  * ---------+---------------+-----------------------------------------
298  * * | all | Use the next argument as the field width.
299  * | | If negative, left-justify the result. If the
300  * | | asterisk is followed by a number and a dollar
301  * | | sign, use the indicated argument as the width.
302  *
303  * Examples of flags:
304  *
305  * # `+' and space flag specifies the sign of non-negative numbers.
306  * sprintf("%d", 123) #=> "123"
307  * sprintf("%+d", 123) #=> "+123"
308  * sprintf("% d", 123) #=> " 123"
309  *
310  * # `#' flag for `o' increases number of digits to show `0'.
311  * # `+' and space flag changes format of negative numbers.
312  * sprintf("%o", 123) #=> "173"
313  * sprintf("%#o", 123) #=> "0173"
314  * sprintf("%+o", -123) #=> "-173"
315  * sprintf("%o", -123) #=> "..7605"
316  * sprintf("%#o", -123) #=> "..7605"
317  *
318  * # `#' flag for `x' add a prefix `0x' for non-zero numbers.
319  * # `+' and space flag disables complements for negative numbers.
320  * sprintf("%x", 123) #=> "7b"
321  * sprintf("%#x", 123) #=> "0x7b"
322  * sprintf("%+x", -123) #=> "-7b"
323  * sprintf("%x", -123) #=> "..f85"
324  * sprintf("%#x", -123) #=> "0x..f85"
325  * sprintf("%#x", 0) #=> "0"
326  *
327  * # `#' for `X' uses the prefix `0X'.
328  * sprintf("%X", 123) #=> "7B"
329  * sprintf("%#X", 123) #=> "0X7B"
330  *
331  * # `#' flag for `b' add a prefix `0b' for non-zero numbers.
332  * # `+' and space flag disables complements for negative numbers.
333  * sprintf("%b", 123) #=> "1111011"
334  * sprintf("%#b", 123) #=> "0b1111011"
335  * sprintf("%+b", -123) #=> "-1111011"
336  * sprintf("%b", -123) #=> "..10000101"
337  * sprintf("%#b", -123) #=> "0b..10000101"
338  * sprintf("%#b", 0) #=> "0"
339  *
340  * # `#' for `B' uses the prefix `0B'.
341  * sprintf("%B", 123) #=> "1111011"
342  * sprintf("%#B", 123) #=> "0B1111011"
343  *
344  * # `#' for `e' forces to show the decimal point.
345  * sprintf("%.0e", 1) #=> "1e+00"
346  * sprintf("%#.0e", 1) #=> "1.e+00"
347  *
348  * # `#' for `f' forces to show the decimal point.
349  * sprintf("%.0f", 1234) #=> "1234"
350  * sprintf("%#.0f", 1234) #=> "1234."
351  *
352  * # `#' for `g' forces to show the decimal point.
353  * # It also disables stripping lowest zeros.
354  * sprintf("%g", 123.4) #=> "123.4"
355  * sprintf("%#g", 123.4) #=> "123.400"
356  * sprintf("%g", 123456) #=> "123456"
357  * sprintf("%#g", 123456) #=> "123456."
358  *
359  * The field width is an optional integer, followed optionally by a
360  * period and a precision. The width specifies the minimum number of
361  * characters that will be written to the result for this field.
362  *
363  * Examples of width:
364  *
365  * # padding is done by spaces, width=20
366  * # 0 or radix-1. <------------------>
367  * sprintf("%20d", 123) #=> " 123"
368  * sprintf("%+20d", 123) #=> " +123"
369  * sprintf("%020d", 123) #=> "00000000000000000123"
370  * sprintf("%+020d", 123) #=> "+0000000000000000123"
371  * sprintf("% 020d", 123) #=> " 0000000000000000123"
372  * sprintf("%-20d", 123) #=> "123 "
373  * sprintf("%-+20d", 123) #=> "+123 "
374  * sprintf("%- 20d", 123) #=> " 123 "
375  * sprintf("%020x", -123) #=> "..ffffffffffffffff85"
376  *
377  * For
378  * numeric fields, the precision controls the number of decimal places
379  * displayed. For string fields, the precision determines the maximum
380  * number of characters to be copied from the string. (Thus, the format
381  * sequence <code>%10.10s</code> will always contribute exactly ten
382  * characters to the result.)
383  *
384  * Examples of precisions:
385  *
386  * # precision for `d', 'o', 'x' and 'b' is
387  * # minimum number of digits <------>
388  * sprintf("%20.8d", 123) #=> " 00000123"
389  * sprintf("%20.8o", 123) #=> " 00000173"
390  * sprintf("%20.8x", 123) #=> " 0000007b"
391  * sprintf("%20.8b", 123) #=> " 01111011"
392  * sprintf("%20.8d", -123) #=> " -00000123"
393  * sprintf("%20.8o", -123) #=> " ..777605"
394  * sprintf("%20.8x", -123) #=> " ..ffff85"
395  * sprintf("%20.8b", -11) #=> " ..110101"
396  *
397  * # "0x" and "0b" for `#x' and `#b' is not counted for
398  * # precision but "0" for `#o' is counted. <------>
399  * sprintf("%#20.8d", 123) #=> " 00000123"
400  * sprintf("%#20.8o", 123) #=> " 00000173"
401  * sprintf("%#20.8x", 123) #=> " 0x0000007b"
402  * sprintf("%#20.8b", 123) #=> " 0b01111011"
403  * sprintf("%#20.8d", -123) #=> " -00000123"
404  * sprintf("%#20.8o", -123) #=> " ..777605"
405  * sprintf("%#20.8x", -123) #=> " 0x..ffff85"
406  * sprintf("%#20.8b", -11) #=> " 0b..110101"
407  *
408  * # precision for `e' is number of
409  * # digits after the decimal point <------>
410  * sprintf("%20.8e", 1234.56789) #=> " 1.23456789e+03"
411  *
412  * # precision for `f' is number of
413  * # digits after the decimal point <------>
414  * sprintf("%20.8f", 1234.56789) #=> " 1234.56789000"
415  *
416  * # precision for `g' is number of
417  * # significant digits <------->
418  * sprintf("%20.8g", 1234.56789) #=> " 1234.5679"
419  *
420  * # <------->
421  * sprintf("%20.8g", 123456789) #=> " 1.2345679e+08"
422  *
423  * # precision for `s' is
424  * # maximum number of characters <------>
425  * sprintf("%20.8s", "string test") #=> " string t"
426  *
427  * Examples:
428  *
429  * sprintf("%d %04x", 123, 123) #=> "123 007b"
430  * sprintf("%08b '%4s'", 123, 123) #=> "01111011 ' 123'"
431  * sprintf("%1$*2$s %2$d %1$s", "hello", 8) #=> " hello 8 hello"
432  * sprintf("%1$*2$s %2$d", "hello", -8) #=> "hello -8"
433  * sprintf("%+g:% g:%-g", 1.23, 1.23, 1.23) #=> "+1.23: 1.23:1.23"
434  * sprintf("%u", -123) #=> "-123"
435  *
436  * For more complex formatting, Ruby supports a reference by name.
437  * %<name>s style uses format style, but %{name} style doesn't.
438  *
439  * Examples:
440  * sprintf("%<foo>d : %<bar>f", { :foo => 1, :bar => 2 })
441  * #=> 1 : 2.000000
442  * sprintf("%{foo}f", { :foo => 1 })
443  * # => "1f"
444  */

606  CHECKNAMEARG(start, len, enc);
607  get_hash(&hash, argc, argv);
608  sym = rb_check_symbol_cstr(start + 1,
609  len - 2 /* without parenthesis */,
610  enc);
611  if (sym != Qnil) nextvalue = rb_hash_lookup2(hash, sym, Qundef);
612  if (nextvalue == Qundef) {
613  rb_enc_raise(enc, rb_eKeyError, "key%.*s not found", len, start);
614  }
615  if (term == '}') goto format_s;
616  p++;
617  goto retry;
618  }
619 
620  case '*':
621  CHECK_FOR_WIDTH(flags);
622  flags |= FWIDTH;
623  GETASTER(width);
624  if (width < 0) {
625  flags |= FMINUS;
626  width = -width;
627  }
628  p++;
629  goto retry;
630 
631  case '.':
632  if (flags & FPREC0) {
633  rb_raise(rb_eArgError, "precision given twice");
634  }
635  flags |= FPREC|FPREC0;
636 
637  prec = 0;
638  p++;
639  if (*p == '*') {
640  GETASTER(prec);
641  if (prec < 0) { /* ignore negative precision */
642  flags &= ~FPREC;
643  }
644  p++;
645  goto retry;
646  }
647 
648  GETNUM(prec, precision);
649  goto retry;
650 
651  case '\n':
652  case '\0':
653  p--;
654  case '%':
655  if (flags != FNONE) {
656  rb_raise(rb_eArgError, "invalid format character - %%");
657  }
658  PUSH("%", 1);
659  break;
660 
661  case 'c':
662  {
663  VALUE val = GETARG();
664  VALUE tmp;
665  unsigned int c;
666  int n;
667 
668  tmp = rb_check_string_type(val);
669  if (!NIL_P(tmp)) {
670  if (rb_enc_strlen(RSTRING_PTR(tmp),RSTRING_END(tmp),enc) != 1) {
671  rb_raise(rb_eArgError, "%%c requires a character");
672  }
673  c = rb_enc_codepoint_len(RSTRING_PTR(tmp), RSTRING_END(tmp), &n, enc);
674  RB_GC_GUARD(tmp);
675  }
676  else {
677  c = NUM2INT(val);
678  n = rb_enc_codelen(c, enc);
679  }
680  if (n <= 0) {
681  rb_raise(rb_eArgError, "invalid character");
682  }
683  if (!(flags & FWIDTH)) {
684  CHECK(n);
685  rb_enc_mbcput(c, &buf[blen], enc);
686  blen += n;
687  }
688  else if ((flags & FMINUS)) {
689  CHECK(n);
690  rb_enc_mbcput(c, &buf[blen], enc);
691  blen += n;
692  FILL(' ', width-1);
693  }
694  else {
695  FILL(' ', width-1);
696  CHECK(n);
697  rb_enc_mbcput(c, &buf[blen], enc);
698  blen += n;
699  }
700  }
701  break;
702 
703  case 's':
704  case 'p':
705  format_s:
706  {
707  VALUE arg = GETARG();
708  long len, slen;
709 
710  if (*p == 'p') arg = rb_inspect(arg);
711  str = rb_obj_as_string(arg);
712  if (OBJ_TAINTED(str)) tainted = 1;
713  len = RSTRING_LEN(str);
714  rb_str_set_len(result, blen);
715  if (coderange != ENC_CODERANGE_BROKEN && scanned < blen) {
716  int cr = coderange;
717  scanned += rb_str_coderange_scan_restartable(buf+scanned, buf+blen, enc, &cr);
718  ENC_CODERANGE_SET(result,
719  (cr == ENC_CODERANGE_UNKNOWN ?
720  ENC_CODERANGE_BROKEN : (coderange = cr)));
721  }
722  enc = rb_enc_check(result, str);
723  if (flags&(FPREC|FWIDTH)) {
724  slen = rb_enc_strlen(RSTRING_PTR(str),RSTRING_END(str),enc);
725  if (slen < 0) {
726  rb_raise(rb_eArgError, "invalid mbstring sequence");
727  }
728  if ((flags&FPREC) && (prec < slen)) {
729  char *p = rb_enc_nth(RSTRING_PTR(str), RSTRING_END(str),
730  prec, enc);
731  slen = prec;
732  len = p - RSTRING_PTR(str);
733  }
734  /* need to adjust multi-byte string pos */
735  if ((flags&FWIDTH) && (width > slen)) {
736  width -= (int)slen;
737  if (!(flags&FMINUS)) {
738  CHECK(width);
739  while (width--) {
740  buf[blen++] = ' ';
741  }
742  }
743  CHECK(len);
744  memcpy(&buf[blen], RSTRING_PTR(str), len);
745  RB_GC_GUARD(str);
746  blen += len;
747  if (flags&FMINUS) {
748  CHECK(width);
749  while (width--) {
750  buf[blen++] = ' ';
751  }
752  }
753  rb_enc_associate(result, enc);
754  break;
755  }
756  }
757  PUSH(RSTRING_PTR(str), len);
758  RB_GC_GUARD(str);
759  rb_enc_associate(result, enc);
760  }
761  break;
762 
763  case 'd':
764  case 'i':
765  case 'o':
766  case 'x':
767  case 'X':
768  case 'b':
769  case 'B':
770  case 'u':
771  {
772  volatile VALUE val = GETARG();
773  int valsign;
774  char nbuf[64], *s;
775  const char *prefix = 0;
776  int sign = 0, dots = 0;
777  char sc = 0;
778  long v = 0;
779  int base, bignum = 0;
780  int len;
781 
782  switch (*p) {
783  case 'd':
784  case 'i':
785  case 'u':
786  sign = 1; break;
787  case 'o':
788  case 'x':
789  case 'X':
790  case 'b':
791  case 'B':
792  if (flags&(FPLUS|FSPACE)) sign = 1;
793  break;
794  }
795  if (flags & FSHARP) {
796  switch (*p) {
797  case 'o':
798  prefix = "0"; break;
799  case 'x':
800  prefix = "0x"; break;
801  case 'X':
802  prefix = "0X"; break;
803  case 'b':
804  prefix = "0b"; break;
805  case 'B':
806  prefix = "0B"; break;
807  }
808  }
809 
810  bin_retry:
811  switch (TYPE(val)) {
812  case T_FLOAT:
813  if (FIXABLE(RFLOAT_VALUE(val))) {
814  val = LONG2FIX((long)RFLOAT_VALUE(val));
815  goto bin_retry;
816  }
817  val = rb_dbl2big(RFLOAT_VALUE(val));
818  if (FIXNUM_P(val)) goto bin_retry;
819  bignum = 1;
820  break;
821  case T_STRING:
822  val = rb_str_to_inum(val, 0, TRUE);
823  goto bin_retry;
824  case T_BIGNUM:
825  bignum = 1;
826  break;
827  case T_FIXNUM:
828  v = FIX2LONG(val);
829  break;
830  default:
831  val = rb_Integer(val);
832  goto bin_retry;
833  }
834 
835  switch (*p) {
836  case 'o':
837  base = 8; break;
838  case 'x':
839  case 'X':
840  base = 16; break;
841  case 'b':
842  case 'B':
843  base = 2; break;
844  case 'u':
845  case 'd':
846  case 'i':
847  default:
848  base = 10; break;
849  }
850 
851  if (base != 10) {
852  int numbits = ffs(base)-1;
853  size_t abs_nlz_bits;
854  size_t numdigits = rb_absint_numwords(val, numbits, &abs_nlz_bits);
855  long i;
856  if (INT_MAX-1 < numdigits) /* INT_MAX is used because rb_long2int is used later. */
857  rb_raise(rb_eArgError, "size too big");
858  if (sign) {
859  if (numdigits == 0)
860  numdigits = 1;
861  tmp = rb_str_new(NULL, numdigits);
862  valsign = rb_integer_pack(val, RSTRING_PTR(tmp), RSTRING_LEN(tmp),
863  1, CHAR_BIT-numbits, INTEGER_PACK_BIG_ENDIAN);
864  for (i = 0; i < RSTRING_LEN(tmp); i++)
865  RSTRING_PTR(tmp)[i] = ruby_digitmap[((unsigned char *)RSTRING_PTR(tmp))[i]];
866  s = RSTRING_PTR(tmp);
867  if (valsign < 0) {
868  sc = '-';
869  width--;
870  }
871  else if (flags & FPLUS) {
872  sc = '+';
873  width--;
874  }
875  else if (flags & FSPACE) {
876  sc = ' ';
877  width--;
878  }
879  }
880  else {
881  /* Following conditional "numdigits++" guarantees the
882  * most significant digit as
883  * - '1'(bin), '7'(oct) or 'f'(hex) for negative numbers
884  * - '0' for zero
885  * - not '0' for positive numbers.
886  *
887  * It also guarantees the most significant two
888  * digits will not be '11'(bin), '77'(oct), 'ff'(hex)
889  * or '00'. */
890  if (numdigits == 0 ||
891  ((abs_nlz_bits != (size_t)(numbits-1) ||
892  !rb_absint_singlebit_p(val)) &&
893  (!bignum ? v < 0 : BIGNUM_NEGATIVE_P(val))))
894  numdigits++;
895  tmp = rb_str_new(NULL, numdigits);
896  valsign = rb_integer_pack(val, RSTRING_PTR(tmp), RSTRING_LEN(tmp),
897  1, CHAR_BIT-numbits, INTEGER_PACK_2COMP | INTEGER_PACK_BIG_ENDIAN);
898  for (i = 0; i < RSTRING_LEN(tmp); i++)
899  RSTRING_PTR(tmp)[i] = ruby_digitmap[((unsigned char *)RSTRING_PTR(tmp))[i]];
900  s = RSTRING_PTR(tmp);
901  dots = valsign < 0;
902  }
903  len = rb_long2int(RSTRING_END(tmp) - s);
904  }
905  else if (!bignum) {
906  valsign = 1;
907  if (v < 0) {
908  v = -v;
909  sc = '-';
910  width--;
911  valsign = -1;
912  }
913  else if (flags & FPLUS) {
914  sc = '+';
915  width--;
916  }
917  else if (flags & FSPACE) {
918  sc = ' ';
919  width--;
920  }
921  snprintf(nbuf, sizeof(nbuf), "%ld", v);
922  s = nbuf;
923  len = (int)strlen(s);
924  }
925  else {
926  tmp = rb_big2str(val, 10);
927  s = RSTRING_PTR(tmp);
928  valsign = 1;
929  if (s[0] == '-') {
930  s++;
931  sc = '-';
932  width--;
933  valsign = -1;
934  }
935  else if (flags & FPLUS) {
936  sc = '+';
937  width--;
938  }
939  else if (flags & FSPACE) {
940  sc = ' ';
941  width--;
942  }
943  len = rb_long2int(RSTRING_END(tmp) - s);
944  }
945 
946  if (dots) {
947  prec -= 2;
948  width -= 2;
949  }
950 
951  if (*p == 'X') {
952  char *pp = s;
953  int c;
954  while ((c = (int)(unsigned char)*pp) != 0) {
955  *pp = rb_enc_toupper(c, enc);
956  pp++;
957  }
958  }
959  if (prefix && !prefix[1]) { /* octal */
960  if (dots) {
961  prefix = 0;
962  }
963  else if (len == 1 && *s == '0') {
964  len = 0;
965  if (flags & FPREC) prec--;
966  }
967  else if ((flags & FPREC) && (prec > len)) {
968  prefix = 0;
969  }
970  }
971  else if (len == 1 && *s == '0') {
972  prefix = 0;
973  }
974  if (prefix) {
975  width -= (int)strlen(prefix);
976  }
977  if ((flags & (FZERO|FMINUS|FPREC)) == FZERO) {
978  prec = width;
979  width = 0;
980  }
981  else {
982  if (prec < len) {
983  if (!prefix && prec == 0 && len == 1 && *s == '0') len = 0;
984  prec = len;
985  }
986  width -= prec;
987  }
988  if (!(flags&FMINUS)) {
989  CHECK(width);
990  while (width-- > 0) {
991  buf[blen++] = ' ';
992  }
993  }
994  if (sc) PUSH(&sc, 1);
995  if (prefix) {
996  int plen = (int)strlen(prefix);
997  PUSH(prefix, plen);
998  }
999  CHECK(prec - len);
1000  if (dots) PUSH("..", 2);
1001  if (!sign && valsign < 0) {
1002  char c = sign_bits(base, p);
1003  while (len < prec--) {
1004  buf[blen++] = c;
1005  }
1006  }
1007  else if ((flags & (FMINUS|FPREC)) != FMINUS) {
1008  while (len < prec--) {
1009  buf[blen++] = '0';
1010  }
1011  }
1012  PUSH(s, len);
1013  RB_GC_GUARD(tmp);
1014  CHECK(width);
1015  while (width-- > 0) {
1016  buf[blen++] = ' ';
1017  }
1018  }
1019  break;
1020 
1021  case 'f':
1022  {
1023  VALUE val = GETARG(), num, den;
1024  int sign = (flags&FPLUS) ? 1 : 0, zero = 0;
1025  long len, done = 0;
1026  int prefix = 0;
1027  if (!RB_TYPE_P(val, T_RATIONAL)) {
1028  nextvalue = val;
1029  goto float_value;
1030  }
1031  if (!(flags&FPREC)) prec = default_float_precision;
1032  den = rb_rational_den(val);
1033  num = rb_rational_num(val);
1034  if (FIXNUM_P(num)) {
1035  if ((SIGNED_VALUE)num < 0) {
1036  long n = -FIX2LONG(num);
1037  num = LONG2FIX(n);
1038  sign = -1;
1039  }
1040  }
1041  else if (rb_num_negative_p(num)) {
1042  sign = -1;
1043  num = rb_funcallv(num, idUMinus, 0, 0);
1044  }
1045  if (den != INT2FIX(1) || prec > 1) {
1046  const ID idDiv = rb_intern("div");
1047  VALUE p10 = rb_int_positive_pow(10, prec);
1048  VALUE den_2 = rb_funcall(den, idDiv, 1, INT2FIX(2));
1049  num = rb_funcallv(num, '*', 1, &p10);
1050  num = rb_funcallv(num, '+', 1, &den_2);
1051  num = rb_funcallv(num, idDiv, 1, &den);
1052  }
1053  else if (prec >= 0) {
1054  zero = prec;
1055  }
1056  val = rb_obj_as_string(num);
1057  len = RSTRING_LEN(val) + zero;
1058  if (prec >= len) ++len; /* integer part 0 */
1059  if (sign || (flags&FSPACE)) ++len;
1060  if (prec > 0) ++len; /* period */
1061  CHECK(len > width ? len : width);
1062  if (sign || (flags&FSPACE)) {
1063  buf[blen++] = sign > 0 ? '+' : sign < 0 ? '-' : ' ';
1064  prefix++;
1065  done++;
1066  }
1067  len = RSTRING_LEN(val) + zero;
1068  t = RSTRING_PTR(val);
1069  if (len > prec) {
1070  memcpy(&buf[blen], t, len - prec);
1071  blen += len - prec;
1072  done += len - prec;
1073  }
1074  else {
1075  buf[blen++] = '0';
1076  done++;
1077  }
1078  if (prec > 0) {
1079  buf[blen++] = '.';
1080  done++;
1081  }
1082  if (zero) {
1083  FILL('0', zero);
1084  done += zero;
1085  }
1086  else if (prec > len) {
1087  FILL('0', prec - len);
1088  memcpy(&buf[blen], t, len);
1089  blen += len;
1090  done += prec;
1091  }
1092  else if (prec > 0) {
1093  memcpy(&buf[blen], t + len - prec, prec);
1094  blen += prec;
1095  done += prec;
1096  }
1097  if ((flags & FWIDTH) && width > done) {
1098  if (!(flags&FMINUS)) {
1099  long i, shifting = (flags&FZERO) ? done - prefix : done;
1100  for (i = 1; i <= shifting; i++)
1101  buf[width - i] = buf[done - i];
1102  blen -= shifting;
1103  FILL((flags&FZERO) ? '0' : ' ', width - done);
1104  blen += shifting;
1105  } else {
1106  FILL(' ', width - done);
1107  }
1108  }
1109  RB_GC_GUARD(val);
1110  break;
1111  }
1112  case 'g':
1113  case 'G':
1114  case 'e':
1115  case 'E':
1116  /* TODO: rational support */
1117  case 'a':
1118  case 'A':
1119  float_value:
1120  {
1121  VALUE val = GETARG();
1122  double fval;
1123  int i, need;
1124  char fbuf[32];
1125 
1126  fval = RFLOAT_VALUE(rb_Float(val));
1127  if (isnan(fval) || isinf(fval)) {
1128  const char *expr;
1129 
1130  if (isnan(fval)) {
1131  expr = "NaN";
1132  }
1133  else {
1134  expr = "Inf";
1135  }
1136  need = (int)strlen(expr);
1137  if ((!isnan(fval) && fval < 0.0) || (flags & FPLUS))
1138  need++;
1139  if ((flags & FWIDTH) && need < width)
1140  need = width;
1141 
1142  CHECK(need + 1);
1143  snprintf(&buf[blen], need + 1, "%*s", need, "");
1144  if (flags & FMINUS) {
1145  if (!isnan(fval) && fval < 0.0)
1146  buf[blen++] = '-';
1147  else if (flags & FPLUS)
1148  buf[blen++] = '+';
1149  else if (flags & FSPACE)
1150  blen++;
1151  memcpy(&buf[blen], expr, strlen(expr));
1152  }
1153  else {
1154  if (!isnan(fval) && fval < 0.0)
1155  buf[blen + need - strlen(expr) - 1] = '-';
1156  else if (flags & FPLUS)
1157  buf[blen + need - strlen(expr) - 1] = '+';
1158  else if ((flags & FSPACE) && need > width)
1159  blen++;
1160  memcpy(&buf[blen + need - strlen(expr)], expr,
1161  strlen(expr));
1162  }
1163  blen += strlen(&buf[blen]);
1164  break;
1165  }
1166 
1167  fmt_setup(fbuf, sizeof(fbuf), *p, flags, width, prec);
1168  need = 0;
1169  if (*p != 'e' && *p != 'E') {
1170  i = INT_MIN;
1171  frexp(fval, &i);
1172  if (i > 0)
1173  need = BIT_DIGITS(i);
1174  }
1175  need += (flags&FPREC) ? prec : default_float_precision;
1176  if ((flags&FWIDTH) && need < width)
1177  need = width;
1178  need += 20;
1179 
1180  CHECK(need);
1181  snprintf(&buf[blen], need, fbuf, fval);
1182  blen += strlen(&buf[blen]);
1183  }
1184  break;
1185  }
1186  flags = FNONE;
1187  }
1188 
1189  sprint_exit:
1190  RB_GC_GUARD(fmt);
1191  /* XXX - We cannot validate the number of arguments if (digit)$ style used.
1192  */
1193  if (posarg >= 0 && nextarg < argc) {
1194  const char *mesg = "too many arguments for format string";
1195  if (RTEST(ruby_debug)) rb_raise(rb_eArgError, "%s", mesg);
1196  if (RTEST(ruby_verbose)) rb_warn("%s", mesg);
1197  }
1198  rb_str_resize(result, blen);
1199 
1200  if (tainted) OBJ_TAINT(result);
1201  return result;
1202 }
