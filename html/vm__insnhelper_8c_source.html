819 static VALUE
820 vm_throw_start(rb_thread_t * const th, rb_control_frame_t * const reg_cfp, int state, const int flag, const rb_num_t level, const VALUE throwobj)
821 {
822  rb_control_frame_t *escape_cfp = NULL;
823  const rb_control_frame_t * const eocfp = RUBY_VM_END_CONTROL_FRAME(th); /* end of control frame pointer */
824 
825  if (flag != 0) {
826  /* do nothing */
827  }
828  else if (state == TAG_BREAK) {
829  int is_orphan = 1;
830  VALUE *ep = GET_EP();
831  rb_iseq_t *base_iseq = GET_ISEQ();
832  escape_cfp = reg_cfp;
833 
834  while (base_iseq->type != ISEQ_TYPE_BLOCK) {
835  if (escape_cfp->iseq->type == ISEQ_TYPE_CLASS) {
836  escape_cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(escape_cfp);
837  ep = escape_cfp->ep;
838  base_iseq = escape_cfp->iseq;
839  }
840  else {
841  ep = VM_EP_PREV_EP(ep);
842  base_iseq = base_iseq->parent_iseq;
843  escape_cfp = rb_vm_search_cf_from_ep(th, escape_cfp, ep);
844  VM_ASSERT(escape_cfp->iseq == base_iseq);
845  }
846  }
847 
848  if (VM_FRAME_TYPE(escape_cfp) == VM_FRAME_MAGIC_LAMBDA) {
849  /* lambda{... break ...} */
850  is_orphan = 0;
851  state = TAG_RETURN;
852  }
853  else {
854  ep = VM_EP_PREV_EP(ep);
855 
856  while (escape_cfp < eocfp) {
857  if (escape_cfp->ep == ep) {
858  const VALUE epc = escape_cfp->pc - escape_cfp->iseq->iseq_encoded;
859  const rb_iseq_t * const iseq = escape_cfp->iseq;
860  const struct iseq_catch_table * const ct = iseq->catch_table;
861  const int ct_size = ct->size;
862  int i;
863 
864  for (i=0; i<ct_size; i++) {
865  const struct iseq_catch_table_entry * const entry = &ct->entries[i];;
866 
867  if (entry->type == CATCH_TYPE_BREAK && entry->start < epc && entry->end >= epc) {
868  if (entry->cont == epc) { /* found! */
869  is_orphan = 0;
870  }
871  break;
872  }
873  }
874  break;
875  }
876 
877  escape_cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(escape_cfp);
878  }
879  }
880 
881  if (is_orphan) {
882  rb_vm_localjump_error("break from proc-closure", throwobj, TAG_BREAK);
883  }
884  }
885  else if (state == TAG_RETRY) {
886  rb_num_t i;
887  VALUE *ep = VM_EP_PREV_EP(GET_EP());
888 
889  for (i = 0; i < level; i++) {
890  ep = VM_EP_PREV_EP(ep);
891  }
892 
893  escape_cfp = rb_vm_search_cf_from_ep(th, reg_cfp, ep);
894  }
895  else if (state == TAG_RETURN) {
896  VALUE *current_ep = GET_EP();
897  VALUE *target_lep = VM_EP_LEP(current_ep);
898  int in_class_frame = 0;
899  escape_cfp = reg_cfp;
900 
901  while (escape_cfp < eocfp) {
902  VALUE *lep = VM_CF_LEP(escape_cfp);
903 
904  if (!target_lep) {
905  target_lep = lep;
906  }
907 
908  if (lep == target_lep && escape_cfp->iseq->type == ISEQ_TYPE_CLASS) {
909  in_class_frame = 1;
910  target_lep = 0;
911  }
912 
913  if (lep == target_lep) {
914  if (VM_FRAME_TYPE(escape_cfp) == VM_FRAME_MAGIC_LAMBDA) {
915  if (in_class_frame) {
916  /* lambda {class A; ... return ...; end} */
917  goto valid_return;
918  }
919  else {
920  VALUE *tep = current_ep;
921 
922  while (target_lep != tep) {
923  if (escape_cfp->ep == tep) {
924  /* in lambda */
925  goto valid_return;
926  }
927  tep = VM_EP_PREV_EP(tep);
928  }
929  }
930  }
931  }
932 
933  if (escape_cfp->ep == target_lep && escape_cfp->iseq->type == ISEQ_TYPE_METHOD) {
934  goto valid_return;
935  }
936 
937  escape_cfp = RUBY_VM_PREVIOUS_CONTROL_FRAME(escape_cfp);
938  }
939  rb_vm_localjump_error("unexpected return", throwobj, TAG_RETURN);
940 
941  valid_return:;
942  /* do nothing */
943  }
944  else {
945  rb_bug("isns(throw): unsupport throw type");
946  }
947 
948  th->state = state;
949  return (VALUE)THROW_DATA_NEW(throwobj, escape_cfp, state);
950 }

1011  *bptr-- = Qnil;

1019  *bptr = rb_ary_new();
1020  }

1022  *bptr = rb_ary_new4(len - num, ptr + num);

1024  }
