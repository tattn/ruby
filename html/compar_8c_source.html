60 /*
61  * call-seq:
62  * obj == other -> true or false
63  *
64  * Compares two objects based on the receiver's <code><=></code>
65  * method, returning true if it returns 0. Also returns true if
66  * _obj_ and _other_ are the same object.
67  *
68  * Even if _obj_ <=> _other_ raised an exception, the exception
69  * is ignored and returns false.
70  */

153 /*
154  * call-seq:
155  * obj.between?(min, max) -> true or false
156  *
157  * Returns <code>false</code> if <i>obj</i> <code><=></code>
158  * <i>min</i> is less than zero or if <i>anObject</i> <code><=></code>
159  * <i>max</i> is greater than zero, <code>true</code> otherwise.
160  *
161  * 3.between?(1, 5) #=> true
162  * 6.between?(1, 5) #=> false
163  * 'cat'.between?('ant', 'dog') #=> true

166  */

169 cmp_between(VALUE x, VALUE min, VALUE max)
170 {

172  if (RTEST(cmp_gt(x, max))) return Qfalse;

176 /*
177  * The <code>Comparable</code> mixin is used by classes whose objects
178  * may be ordered. The class must define the <code><=></code> operator,
179  * which compares the receiver against another object, returning -1, 0,
180  * or +1 depending on whether the receiver is less than, equal to, or
181  * greater than the other object. If the other object is not comparable
182  * then the <code><=></code> operator should return nil.
183  * <code>Comparable</code> uses
184  * <code><=></code> to implement the conventional comparison operators
185  * (<code><</code>, <code><=</code>, <code>==</code>, <code>>=</code>,
186  * and <code>></code>) and the method <code>between?</code>.
187  *
188  * class SizeMatters
189  * include Comparable
190  * attr :str
191  * def <=>(other)
192  * str.size <=> other.str.size
193  * end
194  * def initialize(str)
195  * @str = str
196  * end
197  * def inspect
198  * @str
199  * end
200  * end
201  *
202  * s1 = SizeMatters.new("Z")
203  * s2 = SizeMatters.new("YY")
204  * s3 = SizeMatters.new("XXX")
205  * s4 = SizeMatters.new("WWWW")
206  * s5 = SizeMatters.new("VVVVV")
207  *
208  * s1 < s2 #=> true
209  * s4.between?(s1, s3) #=> false
210  * s4.between?(s3, s5) #=> true
211  * [ s3, s2, s5, s4, s1 ].sort #=> [Z, YY, XXX, WWWW, VVVVV]
212  *
213  */
