424 static void
425 backtrace_each(rb_thread_t *th,
426  void (*init)(void *arg, size_t size),
427  void (*iter_iseq)(void *arg, const rb_control_frame_t *cfp),
428  void (*iter_cfunc)(void *arg, const rb_control_frame_t *cfp, ID mid),
429  void *arg)
430 {
431  rb_control_frame_t *last_cfp = th->cfp;
432  rb_control_frame_t *start_cfp = RUBY_VM_END_CONTROL_FRAME(th);
433  rb_control_frame_t *cfp;
434  ptrdiff_t size, i;
435 
436  /* <- start_cfp (end control frame)
437  * top frame (dummy)
438  * top frame (dummy)
439  * top frame <- start_cfp
440  * top frame
441  * ...
442  * 2nd frame <- lev:0
443  * current frame <- th->cfp
444  */
445 
446  start_cfp =
447  RUBY_VM_NEXT_CONTROL_FRAME(
448  RUBY_VM_NEXT_CONTROL_FRAME(start_cfp)); /* skip top frames */
449 
450  if (start_cfp < last_cfp) {
451  size = 0;
452  }
453  else {
454  size = start_cfp - last_cfp + 1;
455  }
456 
457  init(arg, size);
458 
459  /* SDR(); */
460  for (i=0, cfp = start_cfp; i<size; i++, cfp = RUBY_VM_NEXT_CONTROL_FRAME(cfp)) {
461  /* fprintf(stderr, "cfp: %d\n", (rb_control_frame_t *)(th->stack + th->stack_size) - cfp); */
462  if (cfp->iseq) {
463  if (cfp->pc) {
464  iter_iseq(arg, cfp);
465  }
466  }
467  else if (RUBYVM_CFUNC_FRAME_P(cfp)) {
468  const rb_method_entry_t *me = rb_vm_frame_method_entry(cfp);
469  ID mid = me->def->original_id;
470 
471  iter_cfunc(arg, cfp, mid);
472  }
473  }

817 static VALUE
818 vm_backtrace_to_ary(rb_thread_t *th, int argc, const VALUE *argv, int lev_default, int lev_plus, int to_str)
819 {
820  VALUE level, vn;
821  long lev, n;
822  VALUE btval = backtrace_object(th);
823  VALUE r;
824  rb_backtrace_t *bt;
825 
826  GetCoreDataFromValue(btval, rb_backtrace_t, bt);
827 
828  rb_scan_args(argc, argv, "02", &level, &vn);
829 
830  if (argc == 2 && NIL_P(vn)) argc--;
831 
832  switch (argc) {
833  case 0:
834  lev = lev_default + lev_plus;
835  n = bt->backtrace_size - lev;
836  break;
837  case 1:
838  {
839  long beg, len;
840  switch (rb_range_beg_len(level, &beg, &len, bt->backtrace_size - lev_plus, 0)) {
841  case Qfalse:
842  lev = NUM2LONG(level);
843  if (lev < 0) {
844  rb_raise(rb_eArgError, "negative level (%ld)", lev);
845  }
846  lev += lev_plus;
847  n = bt->backtrace_size - lev;
848  break;
849  case Qnil:
850  return Qnil;
851  default:
852  lev = beg + lev_plus;
853  n = len;
854  break;
855  }
856  break;
857  }
858  case 2:
859  lev = NUM2LONG(level);
860  n = NUM2LONG(vn);
861  if (lev < 0) {
862  rb_raise(rb_eArgError, "negative level (%ld)", lev);
863  }
864  if (n < 0) {
865  rb_raise(rb_eArgError, "negative size (%ld)", n);
866  }
867  lev += lev_plus;
868  break;
869  default:
870  lev = n = 0; /* to avoid warning */
871  break;
872  }
873 
874  if (n == 0) {
875  return rb_ary_new();
876  }
877 
878  if (to_str) {
879  r = backtrace_to_str_ary(btval, lev, n);
880  }
881  else {
882  r = backtrace_to_location_ary(btval, lev, n);
883  }
884  RB_GC_GUARD(btval);
885  return r;
886 }

914  * caller(start=1, length=nil) -> array or nil
915  * caller(range) -> array or nil
916  *
917  * Returns the current execution stack---an array containing strings in
918  * the form <code>file:line</code> or <code>file:line: in
919  * `method'</code>.
920  *
921  * The optional _start_ parameter determines the number of initial stack
922  * entries to omit from the top of the stack.
923  *
924  * A second optional +length+ parameter can be used to limit how many entries
925  * are returned from the stack.

928  * current execution stack.
929  *

938  * end
939  * def c(skip)
940  * b(skip)
941  * end
942  * c(0) #=> ["prog:2:in `a'", "prog:5:in `b'", "prog:8:in `c'", "prog:10:in `<main>'"]

944  * c(2) #=> ["prog:8:in `c'", "prog:12:in `<main>'"]
945  * c(3) #=> ["prog:13:in `<main>'"]
