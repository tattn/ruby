331 {

370 static void
371 w_float(double d, struct dump_arg *arg)
372 {
373  char buf[FLOAT_DIG + (DECIMAL_MANT + 7) / 8 + 10];
374 
375  if (isinf(d)) {
376  if (d < 0) w_cstr("-inf", arg);
377  else w_cstr("inf", arg);
378  }
379  else if (isnan(d)) {
380  w_cstr("nan", arg);
381  }
382  else if (d == 0.0) {
383  if (1.0/d < 0) w_cstr("-0", arg);
384  else w_cstr("0", arg);
385  }
386  else {
387  int decpt, sign, digs, len = 0;
388  char *e, *p = ruby_dtoa(d, 0, 0, &decpt, &sign, &e);
389  if (sign) buf[len++] = '-';
390  digs = (int)(e - p);
391  if (decpt < -3 || decpt > digs) {
392  buf[len++] = p[0];
393  if (--digs > 0) buf[len++] = '.';
394  memcpy(buf + len, p + 1, digs);
395  len += digs;
396  len += snprintf(buf + len, sizeof(buf) - len, "e%d", decpt - 1);
397  }
398  else if (decpt > 0) {
399  memcpy(buf + len, p, decpt);
400  len += decpt;
401  if ((digs -= decpt) > 0) {
402  buf[len++] = '.';
403  memcpy(buf + len, p + decpt, digs);
404  len += digs;
405  }
406  }
407  else {
408  buf[len++] = '0';
409  buf[len++] = '.';
410  if (decpt) {
411  memset(buf + len, '0', -decpt);
412  len -= decpt;
413  }
414  memcpy(buf + len, p, digs);
415  len += digs;
416  }
417  xfree(p);
418  w_bytes(buf, len, arg);
419  }

952 /*
953  * call-seq:
954  * dump( obj [, anIO] , limit=-1 ) -> anIO
955  *
956  * Serializes obj and all descendant objects. If anIO is
957  * specified, the serialized data will be written to it, otherwise the
958  * data will be returned as a String. If limit is specified, the
959  * traversal of subobjects will be limited to that depth. If limit is
960  * negative, no checking of depth will be performed.
961  *
962  * class Klass
963  * def initialize(str)
964  * @str = str
965  * end
966  * def say_hello
967  * @str
968  * end
969  * end
970  *
971  * (produces no output)
972  *
973  * o = Klass.new("hello\n")
974  * data = Marshal.dump(o)
975  * obj = Marshal.load(data)
976  * obj.say_hello #=> "hello\n"
977  *
978  * Marshal can't dump following objects:
979  * * anonymous Class/Module.
980  * * objects which are related to system (ex: Dir, File::Stat, IO, File, Socket
981  * and so on)
982  * * an instance of MatchData, Data, Method, UnboundMethod, Proc, Thread,
983  * ThreadGroup, Continuation
984  * * objects which define singleton methods
985  */
986 static VALUE
987 marshal_dump(int argc, VALUE *argv)
988 {
989  VALUE obj, port, a1, a2;
990  int limit = -1;
991  struct dump_arg *arg;
992  VALUE wrapper; /* used to avoid memory leak in case of exception */
993 
994  port = Qnil;
995  rb_scan_args(argc, argv, "12", &obj, &a1, &a2);
996  if (argc == 3) {
997  if (!NIL_P(a2)) limit = NUM2INT(a2);
998  if (NIL_P(a1)) io_needed();
999  port = a1;
1000  }
1001  else if (argc == 2) {
1002  if (FIXNUM_P(a1)) limit = FIX2INT(a1);
1003  else if (NIL_P(a1)) io_needed();
1004  else port = a1;
1005  }
1006  wrapper = TypedData_Make_Struct(rb_cData, struct dump_arg, &dump_arg_data, arg);
1007  arg->dest = 0;
1008  arg->symbols = st_init_numtable();
1009  arg->data = rb_init_identtable();
1010  arg->infection = 0;
1011  arg->compat_tbl = 0;
1012  arg->encodings = 0;
1013  arg->str = rb_str_buf_new(0);
1014  if (!NIL_P(port)) {
1015  if (!rb_respond_to(port, s_write)) {
1016  io_needed();
1017  }
1018  arg->dest = port;
1019  if (rb_check_funcall(port, s_binmode, 0, 0) != Qundef) {
1020  check_dump_arg(arg, s_binmode);
1021  }
1022  }
1023  else {
1024  port = arg->str;
1025  }
1026 
1027  w_byte(MARSHAL_MAJOR, arg);
1028  w_byte(MARSHAL_MINOR, arg);
1029 
1030  w_object(obj, arg, limit);
1031  if (arg->dest) {
1032  rb_io_write(arg->dest, arg->str);
1033  rb_str_resize(arg->str, 0);
1034  }
1035  clear_dump_arg(arg);
1036  RB_GC_GUARD(wrapper);
1037 
1038  return port;
1039 }

2070 /*
2071  * The marshaling library converts collections of Ruby objects into a
2072  * byte stream, allowing them to be stored outside the currently
2073  * active script. This data may subsequently be read and the original
2074  * objects reconstituted.
2075  *
2076  * Marshaled data has major and minor version numbers stored along
2077  * with the object information. In normal use, marshaling can only
2078  * load data written with the same major version number and an equal
2079  * or lower minor version number. If Ruby's ``verbose'' flag is set
2080  * (normally using -d, -v, -w, or --verbose) the major and minor
2081  * numbers must match exactly. Marshal versioning is independent of
2082  * Ruby's version numbers. You can extract the version by reading the
2083  * first two bytes of marshaled data.
2084  *
2085  * str = Marshal.dump("thing")
2086  * RUBY_VERSION #=> "1.9.0"
2087  * str[0].ord #=> 4
2088  * str[1].ord #=> 8
2089  *
2090  * Some objects cannot be dumped: if the objects to be dumped include
2091  * bindings, procedure or method objects, instances of class IO, or
2092  * singleton objects, a TypeError will be raised.
2093  *
2094  * If your class has special serialization needs (for example, if you
2095  * want to serialize in some specific format), or if it contains
2096  * objects that would otherwise not be serializable, you can implement
2097  * your own serialization strategy.
2098  *
2099  * There are two methods of doing this, your object can define either
2100  * marshal_dump and marshal_load or _dump and _load. marshal_dump will take
2101  * precedence over _dump if both are defined. marshal_dump may result in
2102  * smaller Marshal strings.
2103  *
2104  * == Security considerations
2105  *
2106  * By design, Marshal.load can deserialize almost any class loaded into the
2107  * Ruby process. In many cases this can lead to remote code execution if the
2108  * Marshal data is loaded from an untrusted source.
2109  *
2110  * As a result, Marshal.load is not suitable as a general purpose serialization
2111  * format and you should never unmarshal user supplied input or other untrusted
2112  * data.
2113  *
2114  * If you need to deserialize untrusted data, use JSON or another serialization
2115  * format that is only able to load simple, 'primitive' types such as String,
2116  * Array, Hash, etc. Never allow user input to specify arbitrary types to
2117  * deserialize into.
2118  *
2119  * == marshal_dump and marshal_load
2120  *
2121  * When dumping an object the method marshal_dump will be called.
2122  * marshal_dump must return a result containing the information necessary for
2123  * marshal_load to reconstitute the object. The result can be any object.
2124  *
2125  * When loading an object dumped using marshal_dump the object is first
2126  * allocated then marshal_load is called with the result from marshal_dump.
2127  * marshal_load must recreate the object from the information in the result.
2128  *
2129  * Example:
2130  *
2131  * class MyObj
2132  * def initialize name, version, data
2133  * @name = name
2134  * @version = version
2135  * @data = data
2136  * end
2137  *
2138  * def marshal_dump
2139  * [@name, @version]
2140  * end
2141  *
2142  * def marshal_load array
2143  * @name, @version = array
2144  * end
2145  * end
2146  *
2147  * == _dump and _load
2148  *
2149  * Use _dump and _load when you need to allocate the object you're restoring
2150  * yourself.
2151  *
2152  * When dumping an object the instance method _dump is called with an Integer
2153  * which indicates the maximum depth of objects to dump (a value of -1 implies
2154  * that you should disable depth checking). _dump must return a String
2155  * containing the information necessary to reconstitute the object.
2156  *
2157  * The class method _load should take a String and use it to return an object
2158  * of the same class.
2159  *
2160  * Example:
2161  *
2162  * class MyObj
2163  * def initialize name, version, data
2164  * @name = name
2165  * @version = version
2166  * @data = data
2167  * end
2168  *
2169  * def _dump level
2170  * [@name, @version].join ':'
2171  * end
2172  *
2173  * def self._load args
2174  * new(*args.split(':'))
2175  * end
2176  * end
2177  *
2178  * Since Marshal.dump outputs a string you can have _dump return a Marshal
2179  * string which is Marshal.loaded in _load for complex objects.
2180  */
