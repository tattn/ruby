131 static st_index_t
132 rb_any_hash(VALUE a)
133 {
134  VALUE hval;
135  st_index_t hnum;
136 
137  if (SPECIAL_CONST_P(a)) {
138  if (a == Qundef) return 0;
139  if (STATIC_SYM_P(a)) {
140  a >>= (RUBY_SPECIAL_SHIFT + ID_SCOPE_SHIFT);
141  }
142  else if (FLONUM_P(a)) {
143  /* prevent pathological behavior: [Bug #10761] */
144  return rb_dbl_hash(rb_float_value(a));
145  }
146  hnum = rb_objid_hash((st_index_t)a);
147  }
148  else if (BUILTIN_TYPE(a) == T_STRING) {
149  hnum = rb_str_hash(a);
150  }
151  else if (BUILTIN_TYPE(a) == T_SYMBOL) {
152  hnum = rb_objid_hash((st_index_t)a);
153  }
154  else if (BUILTIN_TYPE(a) == T_FLOAT) {
155  return rb_dbl_hash(rb_float_value(a));

157  else {

159  hnum = FIX2LONG(hval);

161  hnum <<= 1;

187 static st_index_t
188 rb_ident_hash(st_data_t n)
189 {
190  /*
191  * This hash function is lightly-tuned for Ruby. Further tuning
192  * should be possible. Notes:
193  *
194  * - (n >> 3) alone is great for heap objects and OK for fixnum,
195  * however symbols perform poorly.
196  * - (n >> (RUBY_SPECIAL_SHIFT+3)) was added to make symbols hash well,
197  * n.b.: +3 to remove ID scope, +1 worked well initially, too
198  * - (n << 3) was finally added to avoid losing bits for fixnums
199  * - avoid expensive modulo instructions, it is currently only
200  * shifts and bitmask operations.
201  * - flonum (on 64-bit) is pathologically bad, mix the actual
202  * float value in, but do not use the float value as-is since
203  * many integers get interpreted as 2.0 or -2.0 [Bug #10761]
204  */
205 #ifdef USE_FLONUM /* RUBY */
206  if (FLONUM_P(n)) {
207  n ^= (st_data_t)rb_float_value(n);
208  }
209 #endif
210 
211  return (st_index_t)((n>>(RUBY_SPECIAL_SHIFT+3)|(n<<3)) ^ (n>>3));
212 }

493 /*
494  * call-seq:
495  * Hash.new -> new_hash
496  * Hash.new(obj) -> new_hash
497  * Hash.new {|hash, key| block } -> new_hash
498  *
499  * Returns a new, empty hash. If this hash is subsequently accessed by
500  * a key that doesn't correspond to a hash entry, the value returned
501  * depends on the style of <code>new</code> used to create the hash. In
502  * the first form, the access returns <code>nil</code>. If
503  * <i>obj</i> is specified, this single object will be used for
504  * all <em>default values</em>. If a block is specified, it will be
505  * called with the hash object and the key, and should return the
506  * default value. It is the block's responsibility to store the value
507  * in the hash if required.
508  *
509  * h = Hash.new("Go Fish")
510  * h["a"] = 100
511  * h["b"] = 200
512  * h["a"] #=> 100
513  * h["c"] #=> "Go Fish"
514  * # The following alters the single default object
515  * h["c"].upcase! #=> "GO FISH"
516  * h["d"] #=> "GO FISH"
517  * h.keys #=> ["a", "b"]
518  *
519  * # While this creates a new default object each time
520  * h = Hash.new { |hash, key| hash[key] = "Go Fish: #{key}" }
521  * h["c"] #=> "Go Fish: c"
522  * h["c"].upcase! #=> "GO FISH: C"
523  * h["d"] #=> "Go Fish: d"
524  * h.keys #=> ["c", "d"]
525  *
526  */

778 /*
779  * call-seq:
780  * hsh.fetch(key [, default] ) -> obj
781  * hsh.fetch(key) {| key | block } -> obj
782  *
783  * Returns a value from the hash for the given key. If the key can't be
784  * found, there are several options: With no other arguments, it will
785  * raise an <code>KeyError</code> exception; if <i>default</i> is
786  * given, then that will be returned; if the optional code block is
787  * specified, then that will be run and its result returned.
788  *
789  * h = { "a" => 100, "b" => 200 }
790  * h.fetch("a") #=> 100
791  * h.fetch("z", "go fish") #=> "go fish"
792  * h.fetch("z") { |el| "go fish, #{el}"} #=> "go fish, z"
793  *
794  * The following example shows that an exception is raised if the key
795  * is not found and a default value is not supplied.
796  *
797  * h = { "a" => 100, "b" => 200 }
798  * h.fetch("z")
799  *
800  * <em>produces:</em>
801  *
802  * prog.rb:2:in `fetch': key not found (KeyError)
803  * from prog.rb:2
804  *
805  */

807 static VALUE
808 rb_hash_fetch_m(int argc, VALUE *argv, VALUE hash)
809 {
810  VALUE key;
811  st_data_t val;
812  long block_given;
813 
814  rb_check_arity(argc, 1, 2);
815  key = argv[0];
816 
817  block_given = rb_block_given_p();
818  if (block_given && argc == 2) {
819  rb_warn("block supersedes default value argument");
820  }
821  if (!RHASH(hash)->ntbl || !st_lookup(RHASH(hash)->ntbl, key, &val)) {
822  if (block_given) return rb_yield(key);
823  if (argc == 1) {
824  volatile VALUE desc = rb_protect(rb_inspect, key, 0);
825  if (NIL_P(desc)) {
826  desc = rb_any_to_s(key);
827  }
828  desc = rb_str_ellipsize(desc, 65);
829  rb_raise(rb_eKeyError, "key not found: %"PRIsVALUE, desc);
830  }
831  return argv[1];
832  }
833  return (VALUE)val;
834 }

3866 /*
3867  * A Hash is a dictionary-like collection of unique keys and their values.
3868  * Also called associative arrays, they are similar to Arrays, but where an
3869  * Array uses integers as its index, a Hash allows you to use any object
3870  * type.
3871  *
3872  * Hashes enumerate their values in the order that the corresponding keys
3873  * were inserted.
3874  *
3875  * A Hash can be easily created by using its implicit form:
3876  *
3877  * grades = { "Jane Doe" => 10, "Jim Doe" => 6 }
3878  *
3879  * Hashes allow an alternate syntax form when your keys are always symbols.
3880  * Instead of
3881  *
3882  * options = { :font_size => 10, :font_family => "Arial" }
3883  *
3884  * You could write it as:
3885  *
3886  * options = { font_size: 10, font_family: "Arial" }
3887  *
3888  * Each named key is a symbol you can access in hash:
3889  *
3890  * options[:font_size] # => 10
3891  *
3892  * A Hash can also be created through its ::new method:
3893  *
3894  * grades = Hash.new
3895  * grades["Dorothy Doe"] = 9
3896  *
3897  * Hashes have a <em>default value</em> that is returned when accessing
3898  * keys that do not exist in the hash. If no default is set +nil+ is used.
3899  * You can set the default value by sending it as an argument to Hash.new:
3900  *
3901  * grades = Hash.new(0)
3902  *
3903  * Or by using the #default= method:
3904  *
3905  * grades = {"Timmy Doe" => 8}
3906  * grades.default = 0
3907  *
3908  * Accessing a value in a Hash requires using its key:
3909  *
3910  * puts grades["Jane Doe"] # => 0
3911  *
3912  * === Common Uses
3913  *
3914  * Hashes are an easy way to represent data structures, such as
3915  *
3916  * books = {}
3917  * books[:matz] = "The Ruby Language"
3918  * books[:black] = "The Well-Grounded Rubyist"
3919  *
3920  * Hashes are also commonly used as a way to have named parameters in
3921  * functions. Note that no brackets are used below. If a hash is the last
3922  * argument on a method call, no braces are needed, thus creating a really
3923  * clean interface:
3924  *
3925  * Person.create(name: "John Doe", age: 27)
3926  *
3927  * def self.create(params)
3928  * @name = params[:name]
3929  * @age = params[:age]
3930  * end
3931  *
3932  * === Hash Keys
3933  *
3934  * Two objects refer to the same hash key when their <code>hash</code> value
3935  * is identical and the two objects are <code>eql?</code> to each other.
3936  *
3937  * A user-defined class may be used as a hash key if the <code>hash</code>
3938  * and <code>eql?</code> methods are overridden to provide meaningful
3939  * behavior. By default, separate instances refer to separate hash keys.
3940  *
3941  * A typical implementation of <code>hash</code> is based on the
3942  * object's data while <code>eql?</code> is usually aliased to the overridden
3943  * <code>==</code> method:
3944  *
3945  * class Book
3946  * attr_reader :author, :title
3947  *
3948  * def initialize(author, title)
3949  * @author = author
3950  * @title = title
3951  * end
3952  *
3953  * def ==(other)
3954  * self.class === other and
3955  * other.author == @author and
3956  * other.title == @title
3957  * end
3958  *
3959  * alias eql? ==
3960  *
3961  * def hash
3962  * @author.hash ^ @title.hash # XOR
3963  * end
3964  * end
3965  *
3966  * book1 = Book.new 'matz', 'Ruby in a Nutshell'
3967  * book2 = Book.new 'matz', 'Ruby in a Nutshell'
3968  *
3969  * reviews = {}
3970  *
3971  * reviews[book1] = 'Great reference!'
3972  * reviews[book2] = 'Nice and compact!'
3973  *
3974  * reviews.length #=> 1
3975  *
3976  * See also Object#hash and Object#eql?
3977  */
