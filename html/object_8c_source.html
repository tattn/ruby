104 /*
105  * call-seq:
106  * obj == other -> true or false
107  * obj.equal?(other) -> true or false
108  * obj.eql?(other) -> true or false
109  *
110  * Equality --- At the <code>Object</code> level, <code>==</code> returns
111  * <code>true</code> only if +obj+ and +other+ are the same object.
112  * Typically, this method is overridden in descendant classes to provide
113  * class-specific meaning.
114  *
115  * Unlike <code>==</code>, the <code>equal?</code> method should never be
116  * overridden by subclasses as it is used to determine object identity
117  * (that is, <code>a.equal?(b)</code> if and only if <code>a</code> is the
118  * same object as <code>b</code>):
119  *
120  * obj = "a"
121  * other = obj.dup
122  *
123  * obj == other #=> true
124  * obj.equal? other #=> false
125  * obj.equal? obj #=> true
126  *
127  * The <code>eql?</code> method returns <code>true</code> if +obj+ and
128  * +other+ refer to the same hash key. This is used by Hash to test members
129  * for equality. For objects of class <code>Object</code>, <code>eql?</code>
130  * is synonymous with <code>==</code>. Subclasses normally continue this
131  * tradition by aliasing <code>eql?</code> to their overridden <code>==</code>
132  * method, but there are exceptions. <code>Numeric</code> types, for
133  * example, perform type conversion across <code>==</code>, but not across
134  * <code>eql?</code>, so:
135  *
136  * 1 == 1.0 #=> true
137  * 1.eql? 1.0 #=> false
138  */

318  * behavior will be documented under the #+initialize_copy+ method of

322 VALUE
323 rb_obj_clone(VALUE obj)
324 {
325  VALUE clone;
326  VALUE singleton;
327 
328  if (rb_special_const_p(obj)) {
329  rb_raise(rb_eTypeError, "can't clone %s", rb_obj_classname(obj));
330  }
331  clone = rb_obj_alloc(rb_obj_class(obj));
332  RBASIC(clone)->flags &= (FL_TAINT|FL_PROMOTED0|FL_PROMOTED1);
333  RBASIC(clone)->flags |= RBASIC(obj)->flags & ~(FL_PROMOTED0|FL_PROMOTED1|FL_FREEZE|FL_FINALIZE);
334 
335  singleton = rb_singleton_class_clone_and_attach(obj, clone);
336  RBASIC_SET_CLASS(clone, singleton);
337  if (FL_TEST(singleton, FL_SINGLETON)) {
338  rb_singleton_class_attached(singleton, clone);
339  }
340 
341  init_copy(clone, obj);
342  rb_funcall(clone, id_init_clone, 1, obj);
343  RBASIC(clone)->flags |= RBASIC(obj)->flags & FL_FREEZE;
344 
345  return clone;

348 /*
349  * call-seq:
350  * obj.dup -> an_object
351  *
352  * Produces a shallow copy of <i>obj</i>---the instance variables of
353  * <i>obj</i> are copied, but not the objects they reference.
354  * <code>dup</code> copies the tainted state of <i>obj</i>.
355  *
356  * This method may have class-specific behavior. If so, that
357  * behavior will be documented under the #+initialize_copy+ method of
358  * the class.
359  *
360  * === on dup vs clone
361  *
362  * In general, <code>clone</code> and <code>dup</code> may have different
363  * semantics in descendant classes. While <code>clone</code> is used to
364  * duplicate an object, including its internal state, <code>dup</code>
365  * typically uses the class of the descendant object to create the new
366  * instance.
367  *
368  * When using #dup, any modules that the object has been extended with will not
369  * be copied.
370  *
371  * class Klass
372  * attr_accessor :str
373  * end
374  *
375  * module Foo
376  * def foo; 'foo'; end
377  * end
378  *
379  * s1 = Klass.new #=> #<Klass:0x401b3a38>
380  * s1.extend(Foo) #=> #<Klass:0x401b3a38>
381  * s1.foo #=> "foo"
382  *
383  * s2 = s1.clone #=> #<Klass:0x401b3a38>
384  * s2.foo #=> "foo"
385  *
386  * s3 = s1.dup #=> #<Klass:0x401b3a38>
387  * s3.foo #=> NoMethodError: undefined method `foo' for #<Klass:0x401b3a38>
388  *
389  */
