148 ID
149 rb_id_attrset(ID id)
150 {
151  VALUE str, sym;
152  int scope;
153 
154  if (!is_notop_id(id)) {
155  switch (id) {
156  case tAREF: case tASET:
157  return tASET; /* only exception */
158  }
159  rb_name_error(id, "cannot make operator ID :%"PRIsVALUE" attrset",
160  rb_id2str(id));
161  }
162  else {
163  scope = id_type(id);
164  switch (scope) {
165  case ID_LOCAL: case ID_INSTANCE: case ID_GLOBAL:
166  case ID_CONST: case ID_CLASS: case ID_JUNK:
167  break;
168  case ID_ATTRSET:
169  return id;
170  default:
171  {
172  if ((str = lookup_id_str(id)) != 0) {
173  rb_name_error(id, "cannot make unknown type ID %d:%"PRIsVALUE" attrset",
174  scope, str);
175  }
176  else {
177  rb_name_error_str(Qnil, "cannot make unknown type anonymous ID %d:%"PRIxVALUE" attrset",
178  scope, (VALUE)id);
179  }
180  }
181  }
182  }
183 
184  /* make new symbol and ID */
185  if (!(str = lookup_id_str(id))) {
186  static const char id_types[][8] = {
187  "local",
188  "instance",
189  "invalid",
190  "global",
191  "attrset",
192  "const",
193  "class",
194  "junk",
195  };
196  rb_name_error(id, "cannot make anonymous %.*s ID %"PRIxVALUE" attrset",
197  (int)sizeof(id_types[0]), id_types[scope], (VALUE)id);
198  }
199  str = rb_str_dup(str);
200  rb_str_cat(str, "=", 1);
201  sym = lookup_str_sym(str);
202  id = sym ? rb_sym2id(sym) : intern_str(str, 1);
203  return id;
204 }

212 static int
213 is_special_global_name(const char *m, const char *e, rb_encoding *enc)
214 {
215  int mb = 0;
216 
217  if (m >= e) return 0;
218  if (is_global_name_punct(*m)) {
219  ++m;
220  }
221  else if (*m == '-') {
222  if (++m >= e) return 0;
223  if (is_identchar(m, e, enc)) {
224  if (!ISASCII(*m)) mb = 1;
225  m += rb_enc_mbclen(m, e, enc);
226  }
227  }
228  else {
229  if (!rb_enc_isdigit(*m, enc)) return 0;
230  do {
231  if (!ISASCII(*m)) mb = 1;
232  ++m;
233  } while (m < e && rb_enc_isdigit(*m, enc));
234  }
235  return m == e ? mb + 1 : 0;

253 static int
254 rb_enc_symname_type(const char *name, long len, rb_encoding *enc, unsigned int allowed_attrset)
255 {
256  const char *m = name;
257  const char *e = m + len;
258  int type = ID_JUNK;
259 
260  if (!rb_enc_asciicompat(enc)) return -1;
261  if (!m || len <= 0) return -1;
262  switch (*m) {
263  case '\0':
264  return -1;
265 
266  case '$':
267  type = ID_GLOBAL;
268  if (is_special_global_name(++m, e, enc)) return type;
269  goto id;
270 
271  case '@':
272  type = ID_INSTANCE;
273  if (*++m == '@') {
274  ++m;
275  type = ID_CLASS;
276  }
277  goto id;
278 
279  case '<':
280  switch (*++m) {
281  case '<': ++m; break;
282  case '=': if (*++m == '>') ++m; break;
283  default: break;
284  }
285  break;
286 
287  case '>':
288  switch (*++m) {
289  case '>': case '=': ++m; break;
290  }
291  break;
292 
293  case '=':
294  switch (*++m) {
295  case '~': ++m; break;
296  case '=': if (*++m == '=') ++m; break;
297  default: return -1;
298  }
299  break;
300 
301  case '*':
302  if (*++m == '*') ++m;
303  break;
304 
305  case '+': case '-':
306  if (*++m == '@') ++m;
307  break;
308 
309  case '|': case '^': case '&': case '/': case '%': case '~': case '`':
310  ++m;
311  break;
312 
313  case '[':
314  if (m[1] != ']') goto id;
315  ++m;
316  if (*++m == '=') ++m;
317  break;
318 
319  case '!':
320  if (len == 1) return ID_JUNK;
321  switch (*++m) {
322  case '=': case '~': ++m; break;
323  default:
324  if (allowed_attrset & (1U << ID_JUNK)) goto id;
325  return -1;
326  }
327  break;
328 
329  default:
330  type = rb_enc_isupper(*m, enc) ? ID_CONST : ID_LOCAL;
331  id:
332  if (m >= e || (*m != '_' && !rb_enc_isalpha(*m, enc) && ISASCII(*m))) {
333  if (len > 1 && *(e-1) == '=') {
334  type = rb_enc_symname_type(name, len-1, enc, allowed_attrset);
335  if (type != ID_ATTRSET) return ID_ATTRSET;
336  }
337  return -1;
338  }
339  while (m < e && is_identchar(m, e, enc)) m += rb_enc_mbclen(m, e, enc);
340  if (m >= e) break;
341  switch (*m) {
342  case '!': case '?':
343  if (type == ID_GLOBAL || type == ID_CLASS || type == ID_INSTANCE) return -1;
344  type = ID_JUNK;
345  ++m;
346  if (m + 1 < e || *m != '=') break;
347  /* fall through */
348  case '=':
349  if (!(allowed_attrset & (1U << type))) return -1;
350  type = ID_ATTRSET;
351  ++m;
352  break;
353  }
354  break;
355  }
356  return m == e ? type : -1;
357 }

611 static ID
612 intern_str(VALUE str, int mutable)
613 {
614  ID id;
615  ID nid;
616 
617  id = rb_str_symname_type(str, IDSET_ATTRSET_FOR_INTERN);
618  if (id == (ID)-1) id = ID_JUNK;

624  str = rb_str_ellipsize(str, 20);

628  id |= nid;

675  * Returns the <code>Symbol</code> corresponding to <i>str</i>, creating the
676  * symbol if it did not previously exist. See <code>Symbol#id2name</code>.
677  *
678  * "Koala".intern #=> :Koala
679  * s = 'cat'.to_sym #=> :cat
680  * s == :cat #=> true
681  * s = '@cat'.to_sym #=> :@cat
682  * s == :@cat #=> true
683  *
684  * This can also be used to create symbols that cannot be represented using the
685  * <code>:xxx</code> notation.
686  *
687  * 'cat and dog'.to_sym #=> :"cat and dog"
688  */

690 VALUE
691 rb_str_intern(VALUE str)
692 {
693 #if USE_SYMBOL_GC
694  rb_encoding *enc, *ascii;
695  int type;
696 #else
697  ID id;
698 #endif
699  VALUE sym = lookup_str_sym(str);
700 
701  if (sym) {
702  return sym;
703  }
704 
705 #if USE_SYMBOL_GC
706  enc = rb_enc_get(str);
707  ascii = rb_usascii_encoding();
708  if (enc != ascii) {
709  if (sym_check_asciionly(str)) {
710  str = rb_str_dup(str);
711  rb_enc_associate(str, ascii);
712  OBJ_FREEZE(str);
713  enc = ascii;
714  }
715  }
716  str = rb_fstring(str);
717  type = rb_str_symname_type(str, IDSET_ATTRSET_FOR_INTERN);
718  if (type < 0) type = ID_JUNK;
719  return dsymbol_alloc(rb_cSymbol, str, enc, type);
720 #else
721  id = intern_str(str, 0);
722  return ID2SYM(id);
723 #endif
724 }
