373 /*
374  * call-seq:
375  * Process.kill(signal, pid, ...) -> fixnum
376  *
377  * Sends the given signal to the specified process id(s) if _pid_ is positive.
378  * If _pid_ is zero _signal_ is sent to all processes whose group ID is equal
379  * to the group ID of the process. _signal_ may be an integer signal number or
380  * a POSIX signal name (either with or without a +SIG+ prefix). If _signal_ is
381  * negative (or starts with a minus sign), kills process groups instead of
382  * processes. Not all signals are available on all platforms.
383  * The keys and values of +Signal.list+ are known signal names and numbers,
384  * respectively.
385  *
386  * pid = fork do
387  * Signal.trap("HUP") { puts "Ouch!"; exit }
388  * # ... do some work ...
389  * end
390  * # ...
391  * Process.kill("HUP", pid)
392  * Process.wait
393  *
394  * <em>produces:</em>
395  *
396  * Ouch!
397  *
398  * If _signal_ is an integer but wrong for signal,
399  * <code>Errno::EINVAL</code> or +RangeError+ will be raised.
400  * Otherwise unless _signal_ is a +String+ or a +Symbol+, and a known
401  * signal name, +ArgumentError+ will be raised.
402  *
403  * Also, <code>Errno::ESRCH</code> or +RangeError+ for invalid _pid_,
404  * <code>Errno::EPERM</code> when failed because of no privilege,
405  * will be raised. In these cases, signals may have been sent to
406  * preceding processes.
407  */

409 VALUE
410 rb_f_kill(int argc, const VALUE *argv)
411 {
412 #ifndef HAVE_KILLPG
413 #define killpg(pg, sig) kill(-(pg), (sig))
414 #endif
415  int negative = 0;
416  int sig;
417  int i;
418  VALUE str;
419  const char *s;
420 
421  rb_secure(2);
422  rb_check_arity(argc, 2, UNLIMITED_ARGUMENTS);
423 
424  switch (TYPE(argv[0])) {
425  case T_FIXNUM:
426  sig = FIX2INT(argv[0]);
427  break;
428 
429  case T_SYMBOL:
430  str = rb_sym2str(argv[0]);
431  goto str_signal;
432 
433  case T_STRING:
434  str = argv[0];
435  str_signal:
436  s = RSTRING_PTR(str);
437  if (s[0] == '-') {
438  negative++;
439  s++;
440  }
441  if (strncmp(signame_prefix, s, sizeof(signame_prefix)) == 0)
442  s += 3;
443  if ((sig = signm2signo(s)) == 0) {
444  long ofs = s - RSTRING_PTR(str);
445  if (ofs) str = rb_str_subseq(str, ofs, RSTRING_LEN(str)-ofs);
446  rb_raise(rb_eArgError, "unsupported name `SIG%"PRIsVALUE"'", str);
447  }
448 
449  if (negative)
450  sig = -sig;
451  break;
452 
453  default:
454  str = rb_check_string_type(argv[0]);
455  if (!NIL_P(str)) {
456  goto str_signal;
457  }
458  rb_raise(rb_eArgError, "bad signal type %s",
459  rb_obj_classname(argv[0]));
460  break;
461  }
462 
463  if (argc <= 1) return INT2FIX(0);
464 
465  if (sig < 0) {
466  sig = -sig;
467  for (i=1; i<argc; i++) {
468  if (killpg(NUM2PIDT(argv[i]), sig) < 0)
469  rb_sys_fail(0);
470  }
471  }
472  else {
473  const rb_pid_t self = (GET_THREAD() == GET_VM()->main_thread) ? getpid() : -1;
474  int wakeup = 0;
475 
476  for (i=1; i<argc; i++) {
477  rb_pid_t pid = NUM2PIDT(argv[i]);
478 
479  if ((sig != 0) && (self != -1) && (pid == self)) {
480  int t;
481  /*
482  * When target pid is self, many caller assume signal will be
483  * delivered immediately and synchronously.
484  */
485  switch (sig) {
486  case SIGSEGV:
487 #ifdef SIGBUS
488  case SIGBUS:
489 #endif
490 #ifdef SIGKILL
491  case SIGKILL:
492 #endif
493 #ifdef SIGSTOP
494  case SIGSTOP:
495 #endif
496  ruby_kill(pid, sig);
497  break;
498  default:
499  t = signal_ignored(sig);
500  if (t) {
501  if (t < 0 && kill(pid, sig))
502  rb_sys_fail(0);
503  break;
504  }
505  signal_enque(sig);
506  wakeup = 1;
507  }
508  }
509  else if (kill(pid, sig) < 0) {
510  rb_sys_fail(0);
511  }
512  }
513  if (wakeup) {
514  rb_threadptr_check_signal(GET_VM()->main_thread);
515  }
516  }
517  rb_thread_execute_interrupts(rb_thread_current());
518 
519  return INT2FIX(i-1);
520 }

775  const DEFINE_MCONTEXT_PTR(mctx, ctx);
776 # if defined __linux__
777 # if defined REG_RSP
778  const greg_t sp = mctx->gregs[REG_RSP];
779 # else
780  const greg_t sp = mctx->gregs[REG_ESP];
781 # endif
782 # elif defined __APPLE__
783 # if defined(__LP64__)
784  const uintptr_t sp = mctx->__ss.__rsp;
785 # else
786  const uintptr_t sp = mctx->__ss.__esp;
787 # endif
788 # elif defined __FreeBSD__
789 # if defined(__amd64__)
790  const __register_t sp = mctx->mc_rsp;
791 # else
792  const __register_t sp = mctx->mc_esp;
793 # endif
794 # endif
795  enum {pagesize = 4096};
796  const uintptr_t sp_page = (uintptr_t)sp / pagesize;
797  const uintptr_t fault_page = addr / pagesize;
798 
799  /* SP in ucontext is not decremented yet when `push` failed, so
800  * the fault page can be the next. */
801  if (sp_page == fault_page || sp_page == fault_page + 1) {
802  rb_thread_t *th = ruby_current_thread;
803  if ((uintptr_t)th->tag->buf / pagesize == sp_page) {
804  /* drop the last tag if it is close to the fault,
805  * otherwise it can cause stack overflow again at the same
806  * place. */
807  th->tag = th->tag->prev;
808  }
809  clear_received_signal();
810  ruby_thread_stack_overflow(th);
811  }
812 }

939 static void
940 signal_exec(VALUE cmd, int safe, int sig)
941 {
942  rb_thread_t *cur_th = GET_THREAD();
943  volatile unsigned long old_interrupt_mask = cur_th->interrupt_mask;
944  int state;
945 
946  /*
947  * workaround the following race:
948  * 1. signal_enque queues signal for execution
949  * 2. user calls trap(sig, "IGNORE"), setting SIG_IGN
950  * 3. rb_signal_exec runs on queued signal
951  */
952  if (IMMEDIATE_P(cmd))
953  return;
954 
955  cur_th->interrupt_mask |= TRAP_INTERRUPT_MASK;
956  TH_PUSH_TAG(cur_th);
957  if ((state = EXEC_TAG()) == 0) {
958  VALUE signum = INT2NUM(sig);
959  rb_eval_cmd(cmd, rb_ary_new3(1, signum), safe);
960  }
961  TH_POP_TAG();
962  cur_th = GET_THREAD();
963  cur_th->interrupt_mask = old_interrupt_mask;
964 
965  if (state) {
966  /* XXX: should be replaced with rb_threadptr_pending_interrupt_enque() */
967  JUMP_TAG(state);
968  }
969 }

1074 static sighandler_t
1075 trap_handler(VALUE *cmd, int sig)
1076 {
1077  sighandler_t func = sighandler;
1078  VALUE command;
1079 
1080  if (NIL_P(*cmd)) {
1081  func = SIG_IGN;
1082  }
1083  else {
1084  command = rb_check_string_type(*cmd);
1085  if (NIL_P(command) && SYMBOL_P(*cmd)) {
1086  command = rb_sym2str(*cmd);
1087  if (!command) rb_raise(rb_eArgError, "bad handler");
1088  }
1089  if (!NIL_P(command)) {
1090  SafeStringValue(command); /* taint check */
1091  *cmd = command;
1092  switch (RSTRING_LEN(command)) {
1093  case 0:
1094  goto sig_ign;
1095  break;
1096  case 14:
1097  if (strncmp(RSTRING_PTR(command), "SYSTEM_DEFAULT", 14) == 0) {
1098  func = SIG_DFL;
1099  *cmd = 0;
1100  }
1101  break;
1102  case 7:
1103  if (strncmp(RSTRING_PTR(command), "SIG_IGN", 7) == 0) {
1104 sig_ign:
1105  func = SIG_IGN;
1106  *cmd = Qtrue;
1107  }
1108  else if (strncmp(RSTRING_PTR(command), "SIG_DFL", 7) == 0) {
1109 sig_dfl:
1110  func = default_handler(sig);
1111  *cmd = 0;
1112  }
1113  else if (strncmp(RSTRING_PTR(command), "DEFAULT", 7) == 0) {
1114  goto sig_dfl;
1115  }
1116  break;
1117  case 6:
1118  if (strncmp(RSTRING_PTR(command), "IGNORE", 6) == 0) {
1119  goto sig_ign;
1120  }
1121  break;
1122  case 4:
1123  if (strncmp(RSTRING_PTR(command), "EXIT", 4) == 0) {
1124  *cmd = Qundef;
1125  }
1126  break;
1127  }
1128  }
1129  else {
1130  rb_proc_t *proc;
1131  GetProcPtr(*cmd, proc);
1132  (void)proc;
1133  }
1134  }
1135 
1136  return func;
1137 }

1139 static int
1140 trap_signm(VALUE vsig)
1141 {
1142  int sig = -1;
1143  const char *s;
1144 
1145  switch (TYPE(vsig)) {
1146  case T_FIXNUM:
1147  sig = FIX2INT(vsig);
1148  if (sig < 0 || sig >= NSIG) {
1149  rb_raise(rb_eArgError, "invalid signal number (%d)", sig);
1150  }
1151  break;
1152 
1153  case T_SYMBOL:
1154  vsig = rb_sym2str(vsig);
1155  s = RSTRING_PTR(vsig);
1156  goto str_signal;
1157 
1158  default:
1159  s = StringValuePtr(vsig);
1160 
1161  str_signal:
1162  if (strncmp(signame_prefix, s, sizeof(signame_prefix)) == 0)
1163  s += 3;
1164  sig = signm2signo(s);
1165  if (sig == 0 && strcmp(s, "EXIT") != 0) {
1166  long ofs = s - RSTRING_PTR(vsig);
1167  if (ofs) vsig = rb_str_subseq(vsig, ofs, RSTRING_LEN(vsig)-ofs);
1168  rb_raise(rb_eArgError, "unsupported signal SIG%"PRIsVALUE"", vsig);
1169  }
1170  }
1171  return sig;
1172 }

1174 static VALUE
1175 trap(int sig, sighandler_t func, VALUE command)
1176 {
1177  sighandler_t oldfunc;
1178  VALUE oldcmd;
1179  rb_vm_t *vm = GET_VM();
1180 
1181  /*
1182  * Be careful. ruby_signal() and trap_list[sig].cmd must be changed
1183  * atomically. In current implementation, we only need to don't call
1184  * RUBY_VM_CHECK_INTS().
1185  */
1186  if (sig == 0) {
1187  oldfunc = SIG_ERR;
1188  }
1189  else {
1190  oldfunc = ruby_signal(sig, func);
1191  if (oldfunc == SIG_ERR) rb_sys_fail_str(rb_signo2signm(sig));
1192  }
1193  oldcmd = vm->trap_list[sig].cmd;
1194  switch (oldcmd) {
1195  case 0:
1196  case Qtrue:
1197  if (oldfunc == SIG_IGN) oldcmd = rb_str_new2("IGNORE");
1198  else if (oldfunc == SIG_DFL) oldcmd = rb_str_new2("SYSTEM_DEFAULT");
1199  else if (oldfunc == sighandler) oldcmd = rb_str_new2("DEFAULT");
1200  else oldcmd = Qnil;
1201  break;
1202  case Qnil:
1203  break;
1204  case Qundef:
1205  oldcmd = rb_str_new2("EXIT");
1206  break;
1207  }
1208 
1209  vm->trap_list[sig].cmd = command;
1210  vm->trap_list[sig].safe = rb_safe_level();
1211 
1212  return oldcmd;
1213 }

1245 /*
1246  * call-seq:
1247  * Signal.trap( signal, command ) -> obj
1248  * Signal.trap( signal ) {| | block } -> obj
1249  *
1250  * Specifies the handling of signals. The first parameter is a signal
1251  * name (a string such as ``SIGALRM'', ``SIGUSR1'', and so on) or a
1252  * signal number. The characters ``SIG'' may be omitted from the
1253  * signal name. The command or block specifies code to be run when the
1254  * signal is raised.
1255  * If the command is the string ``IGNORE'' or ``SIG_IGN'', the signal
1256  * will be ignored.
1257  * If the command is ``DEFAULT'' or ``SIG_DFL'', the Ruby's default handler
1258  * will be invoked.
1259  * If the command is ``EXIT'', the script will be terminated by the signal.
1260  * If the command is ``SYSTEM_DEFAULT'', the operating system's default
1261  * handler will be invoked.
1262  * Otherwise, the given command or block will be run.
1263  * The special signal name ``EXIT'' or signal number zero will be
1264  * invoked just prior to program termination.
1265  * trap returns the previous handler for the given signal.
1266  *
1267  * Signal.trap(0, proc { puts "Terminating: #{$$}" })
1268  * Signal.trap("CLD") { puts "Child died" }
1269  * fork && Process.wait
1270  *
1271  * produces:
1272  * Terminating: 27461
1273  * Child died
1274  * Terminating: 27460
1275  */
1276 static VALUE
1277 sig_trap(int argc, VALUE *argv)
1278 {
1279  int sig;
1280  sighandler_t func;
1281  VALUE cmd;
1282 
1283  rb_secure(2);
1284  rb_check_arity(argc, 1, 2);
1285 
1286  sig = trap_signm(argv[0]);
1287  if (reserved_signal_p(sig)) {
1288  const char *name = signo2signm(sig);
1289  if (name)
1290  rb_raise(rb_eArgError, "can't trap reserved signal: SIG%s", name);
1291  else
1292  rb_raise(rb_eArgError, "can't trap reserved signal: %d", sig);
1293  }
1294 
1295  if (argc == 1) {
1296  cmd = rb_block_proc();
1297  func = sighandler;
1298  }
1299  else {
1300  cmd = argv[1];
1301  func = trap_handler(&cmd, sig);
1302  }
1303 
1304  if (OBJ_TAINTED(cmd)) {
1305  rb_raise(rb_eSecurityError, "Insecure: tainted signal trap");
1306  }
1307 
1308  return trap(sig, func, cmd);
1309 }

1389  * processes. Some signals have a defined effect on the process, while
1390  * others may be trapped at the code level and acted upon. For
1391  * example, your process may trap the USR1 signal and use it to toggle
1392  * debugging, and may use TERM to initiate a controlled shutdown.
1393  *
1394  * pid = fork do
1395  * Signal.trap("USR1") do
1396  * $debug = !$debug
1397  * puts "Debug now: #$debug"
1398  * end

1400  * puts "Terminating..."
1401  * shutdown()
1402  * end
1403  * # . . . do some work . . .
1404  * end
1405  *
1406  * Process.detach(pid)
1407  *
1408  * # Controlling program:
1409  * Process.kill("USR1", pid)
1410  * # ...
1411  * Process.kill("USR1", pid)
1412  * # ...
1413  * Process.kill("TERM", pid)
1414  *
1415  * produces:
1416  * Debug now: true
1417  * Debug now: false
1418  * Terminating...
1419  *
1420  * The list of available signal names and their interpretation is
1421  * system dependent. Signal delivery semantics may also vary between
1422  * systems; in particular signal delivery may not always be reliable.
