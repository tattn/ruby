336 /*
337  * call-seq:
338  * Struct.new([class_name] [, member_name]+>) -> StructClass
339  * Struct.new([class_name] [, member_name]+>) {|StructClass| block } -> StructClass
340  * StructClass.new(value, ...) -> obj
341  * StructClass[value, ...] -> obj
342  *
343  * The first two forms are used to create a new Struct subclass +class_name+
344  * that can contain a value for each +member_name+. This subclass can be
345  * used to create instances of the structure like any other Class.
346  *
347  * If the +class_name+ is omitted an anonymous structure class will be
348  * created. Otherwise, the name of this struct will appear as a constant in
349  * class Struct, so it must be unique for all Structs in the system and
350  * must start with a capital letter. Assigning a structure class to a
351  * constant also gives the class the name of the constant.
352  *
353  * # Create a structure with a name under Struct
354  * Struct.new("Customer", :name, :address)
355  * #=> Struct::Customer
356  * Struct::Customer.new("Dave", "123 Main")
357  * #=> #<struct Struct::Customer name="Dave", address="123 Main">
358  *
359  * If a block is given it will be evaluated in the context of
360  * +StructClass+, passing the created class as a parameter:
361  *
362  * Customer = Struct.new(:name, :address) do
363  * def greeting
364  * "Hello #{name}!"
365  * end
366  * end
367  * Customer.new("Dave", "123 Main").greeting # => "Hello Dave!"
368  *
369  * This is the recommended way to customize a struct. Subclassing an
370  * anonymous struct creates an extra anonymous class that will never be used.
371  *
372  * The last two forms create a new instance of a struct subclass. The number
373  * of +value+ parameters must be less than or equal to the number of
374  * attributes defined for the structure. Unset parameters default to +nil+.
375  * Passing more parameters than number of attributes will raise
376  * an ArgumentError.
377  *
378  * # Create a structure named by its constant
379  * Customer = Struct.new(:name, :address)
380  * #=> Customer
381  * Customer.new("Dave", "123 Main")
382  * #=> #<struct Customer name="Dave", address="123 Main">
383  */

923 /*
924  * call-seq:
925  * struct == other -> true or false
926  *
927  * Equality---Returns +true+ if +other+ has the same struct subclass and has
928  * equal member values (according to Object#==).
929  *
930  * Customer = Struct.new(:name, :address, :zip)
931  * joe = Customer.new("Joe Smith", "123 Maple, Anytown NC", 12345)
932  * joejr = Customer.new("Joe Smith", "123 Maple, Anytown NC", 12345)
933  * jane = Customer.new("Jane Doe", "456 Elm, Anytown NC", 12345)
934  * joe == joejr #=> true
935  * joe == jane #=> false
936  */
937

939 rb_struct_equal(VALUE s, VALUE s2)
940 {
941  if (s == s2) return Qtrue;
942  if (!RB_TYPE_P(s2, T_STRUCT)) return Qfalse;
943  if (rb_obj_class(s) != rb_obj_class(s2)) return Qfalse;
944  if (RSTRUCT_LEN(s) != RSTRUCT_LEN(s2)) {
945  rb_bug("inconsistent struct"); /* should never happen */
946  }
947 
948  return rb_exec_recursive_paired(recursive_equal, s, s2, s2);
949 }
