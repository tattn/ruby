110 static inline void
111 args_copy(struct args_info *args)
112 {
113  if (args->rest != Qfalse) {
114  int argc = args->argc;
115  args->argc = 0;
116  args->rest = rb_ary_dup(args->rest); /* make dup */
117 
118  /*
119  * argv: [m0, m1, m2, m3]
120  * rest: [a0, a1, a2, a3, a4, a5]
121  * ^
122  * rest_index
123  *
124  * #=> first loop
125  *
126  * argv: [m0, m1]
127  * rest: [m2, m3, a2, a3, a4, a5]
128  * ^
129  * rest_index
130  *
131  * #=> 2nd loop
132  *
133  * argv: [] (argc == 0)
134  * rest: [m0, m1, m2, m3, a2, a3, a4, a5]
135  * ^
136  * rest_index
137  */
138  while (args->rest_index > 0 && argc > 0) {
139  RARRAY_ASET(args->rest, --args->rest_index, args->argv[--argc]);
140  }
141  while (argc > 0) {
142  rb_ary_unshift(args->rest, args->argv[--argc]);
143  }
144  }
145  else if (args->argc > 0) {
146  args->rest = rb_ary_new_from_values(args->argc, args->argv);
147  args->rest_index = 0;
148  args->argc = 0;
149  }
150 }

383 static void
384 args_setup_kw_parameters(VALUE* const passed_values, const int passed_keyword_len, const VALUE *const passed_keywords,
385  const rb_iseq_t * const iseq, VALUE * const locals)
386 {
387  const ID *acceptable_keywords = iseq->param.keyword->table;
388  const int req_key_num = iseq->param.keyword->required_num;
389  const int key_num = iseq->param.keyword->num;
390  const VALUE * const default_values = iseq->param.keyword->default_values;
391  VALUE missing = 0;
392  int i, di, found = 0;
393  int unspecified_bits = 0;
394  VALUE unspecified_bits_value = Qnil;
395 
396  for (i=0; i<req_key_num; i++) {
397  ID key = acceptable_keywords[i];
398  if (args_setup_kw_parameters_lookup(key, &locals[i], passed_keywords, passed_values, passed_keyword_len)) {
399  found++;
400  }
401  else {
402  if (!missing) missing = rb_ary_tmp_new(1);
403  rb_ary_push(missing, ID2SYM(key));
404  }
405  }
406 
407  if (missing) argument_kw_error(GET_THREAD(), iseq, "missing", missing);
408 
409  for (di=0; i<key_num; i++, di++) {
410  if (args_setup_kw_parameters_lookup(acceptable_keywords[i], &locals[i], passed_keywords, passed_values, passed_keyword_len)) {
411  found++;
412  }
413  else {
414  if (default_values[di] == Qundef) {
415  locals[i] = Qnil;
416 
417  if (LIKELY(i < 32)) { /* TODO: 32 -> Fixnum's max bits */
418  unspecified_bits |= 0x01 << di;
419  }
420  else {
421  if (NIL_P(unspecified_bits_value)) {
422  /* fixnum -> hash */
423  int j;
424  unspecified_bits_value = rb_hash_new();
425 
426  for (j=0; j<32; j++) {
427  if (unspecified_bits & (0x01 << j)) {
428  rb_hash_aset(unspecified_bits_value, INT2FIX(j), Qtrue);
429  }
430  }
431  }
432  rb_hash_aset(unspecified_bits_value, INT2FIX(di), Qtrue);
433  }
434  }
435  else {
436  locals[i] = default_values[di];
437  }
438  }
439  }
440 
441  if (iseq->param.flags.has_kwrest) {
442  const int rest_hash_index = key_num + 1;
443  locals[rest_hash_index] = make_unused_kw_hash(passed_keywords, passed_keyword_len, passed_values, FALSE);
444  }
445  else {
446  if (found != passed_keyword_len) {
447  VALUE keys = make_unused_kw_hash(passed_keywords, passed_keyword_len, passed_values, TRUE);
448  argument_kw_error(GET_THREAD(), iseq, "unknown", keys);
449  }
450  }
451 
452  if (NIL_P(unspecified_bits_value)) {
453  unspecified_bits_value = INT2FIX(unspecified_bits);
454  }
455  locals[key_num] = unspecified_bits_value;
456 }

503  VALUE * const locals, const enum arg_setup_type arg_setup_type)
504 {
505  const int min_argc = iseq->param.lead_num + iseq->param.post_num;
506  const int max_argc = (iseq->param.flags.has_rest == FALSE) ? min_argc + iseq->param.opt_num : UNLIMITED_ARGUMENTS;

516  *

519  * <- iseq->param.size------------------>
520  * ^ locals ^ sp
521  *
522  * =>

569  case arg_setup_block:

572  iseq->param.flags.has_kw || iseq->param.flags.has_kwrest) &&

576  }
577  break;

579  if (given_argc == 1 &&
580  given_argc != iseq->param.lead_num &&
581  !iseq->param.flags.has_rest &&

583  given_argc = RARRAY_LENINT(args->rest);
584  }
585  }

588  if (given_argc < min_argc) {
589  if (given_argc == min_argc - 1 && args->kw_argv) {
590  args_stored_kw_argv_to_hash(args);
591  given_argc = args_argc(args);
592  }

595  CHECK_VM_STACK_OVERFLOW(th->cfp, min_argc);

602  }
603  }

614  if (arg_setup_type == arg_setup_block) {
615  /* truncate */
616  args_reduce(args, given_argc - max_argc);

618  }
619  else {
620  argument_arity_error(th, iseq, given_argc, min_argc, max_argc);
621  }
622  }

626  }

630  }
631

633  int opt = args_setup_opt_parameters(args, iseq->param.opt_num, locals + iseq->param.lead_num);

665  }

669  }
