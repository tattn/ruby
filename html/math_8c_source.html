106 static VALUE
107 math_atan2(VALUE obj, VALUE y, VALUE x)
108 {
109 #ifndef M_PI
110 # define M_PI 3.14159265358979323846
111 #endif
112  double dx, dy;
113  dx = Get_Double(x);
114  dy = Get_Double(y);
115  if (dx == 0.0 && dy == 0.0) {
116  if (!signbit(dx))
117  return DBL2NUM(dy);
118  if (!signbit(dy))
119  return DBL2NUM(M_PI);
120  return DBL2NUM(-M_PI);
121  }
122 #ifndef ATAN2_INF_C99
123  if (isinf(dx) && isinf(dy)) {
124  /* optimization for FLONUM */
125  if (dx < 0.0) {
126  const double dz = (3.0 * M_PI / 4.0);
127  return (dy < 0.0) ? DBL2NUM(-dz) : DBL2NUM(dz);
128  }
129  else {
130  const double dz = (M_PI / 4.0);
131  return (dy < 0.0) ? DBL2NUM(-dz) : DBL2NUM(dz);
132  }
133  }
134 #endif
135  return DBL2NUM(atan2(dy, dx));
136 }
