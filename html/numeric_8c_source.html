276 {
277  VALUE ary;

279 
280  a[0] = *x; a[1] = *y;

494 /*
495  * call-seq:
496  * num.divmod(numeric) -> array
497  *
498  * Returns an array containing the quotient and modulus obtained by dividing
499  * +num+ by +numeric+.
500  *
501  * If <code>q, r = * x.divmod(y)</code>, then
502  *
503  * q = floor(x/y)
504  * x = q*y+r
505  *
506  * The quotient is rounded toward -infinity, as shown in the following table:
507  *
508  * a | b | a.divmod(b) | a/b | a.modulo(b) | a.remainder(b)
509  * ------+-----+---------------+---------+-------------+---------------
510  * 13 | 4 | 3, 1 | 3 | 1 | 1
511  * ------+-----+---------------+---------+-------------+---------------
512  * 13 | -4 | -4, -3 | -4 | -3 | 1
513  * ------+-----+---------------+---------+-------------+---------------
514  * -13 | 4 | -4, 3 | -4 | 3 | -1
515  * ------+-----+---------------+---------+-------------+---------------
516  * -13 | -4 | 3, -1 | 3 | -1 | -1
517  * ------+-----+---------------+---------+-------------+---------------
518  * 11.5 | 4 | 2, 3.5 | 2.875 | 3.5 | 3.5
519  * ------+-----+---------------+---------+-------------+---------------
520  * 11.5 | -4 | -3, -0.5 | -2.875 | -0.5 | 3.5
521  * ------+-----+---------------+---------+-------------+---------------
522  * -11.5 | 4 | -3, 0.5 | -2.875 | 0.5 | -3.5
523  * ------+-----+---------------+---------+-------------+---------------
524  * -11.5 | -4 | 2, -3.5 | 2.875 | -3.5 | -3.5
525  *
526  *
527  * Examples
528  *
529  * 11.divmod(3) #=> [3, 2]
530  * 11.divmod(-3) #=> [-4, -1]
531  * 11.divmod(3.5) #=> [3, 0.5]
532  * (-11).divmod(3.5) #=> [-4, 3.0]
533  * (11.5).divmod(3.5) #=> [3, 1.0]
534  */

1792 flo_round(int argc, VALUE *argv, VALUE num)
1793 {
1794  VALUE nd;
1795  double number, f;
1796  int ndigits = 0;
1797  int binexp;
1798  enum {float_dig = DBL_DIG+2};
1799 
1800  if (argc > 0 && rb_scan_args(argc, argv, "01", &nd) == 1) {
1801  ndigits = NUM2INT(nd);
1802  }
1803  if (ndigits < 0) {
1804  return int_round_0(flo_truncate(num), ndigits);
1805  }
1806  number = RFLOAT_VALUE(num);
1807  if (ndigits == 0) {
1808  return dbl2ival(number);
1809  }
1810  frexp(number, &binexp);
1811 
1812 /* Let `exp` be such that `number` is written as:"0.#{digits}e#{exp}",
1813  i.e. such that 10 ** (exp - 1) <= |number| < 10 ** exp
1814  Recall that up to float_dig digits can be needed to represent a double,
1815  so if ndigits + exp >= float_dig, the intermediate value (number * 10 ** ndigits)
1816  will be an integer and thus the result is the original number.
1817  If ndigits + exp <= 0, the result is 0 or "1e#{exp}", so
1818  if ndigits + exp < 0, the result is 0.
1819  We have:
1820  2 ** (binexp-1) <= |number| < 2 ** binexp
1821  10 ** ((binexp-1)/log_2(10)) <= |number| < 10 ** (binexp/log_2(10))
1822  If binexp >= 0, and since log_2(10) = 3.322259:
1823  10 ** (binexp/4 - 1) < |number| < 10 ** (binexp/3)
1824  floor(binexp/4) <= exp <= ceil(binexp/3)
1825  If binexp <= 0, swap the /4 and the /3
1826  So if ndigits + floor(binexp/(4 or 3)) >= float_dig, the result is number
1827  If ndigits + ceil(binexp/(3 or 4)) < 0 the result is 0
1828 */
1829  if (isinf(number) || isnan(number) ||
1830  (ndigits >= float_dig - (binexp > 0 ? binexp / 4 : binexp / 3 - 1))) {

1835  }

1974 {

1976  double n = (end - beg)/unit;
1977  double err = (fabs(beg) + fabs(end) + fabs(end-beg)) / fabs(unit) * epsilon;
1978

1980  return unit > 0 ? beg <= end : beg >= end;
1981  }

1989  n = 0;
1990  else
1991  n = floor(n - err);

1993  else {

2131  * num.step(by: step, to: limit) {|i| block } -> self
2132  * num.step(by: step, to: limit) -> an_enumerator

2134  * num.step(limit=nil, step=1) -> an_enumerator
2135  *
2136  * Invokes the given block with the sequence of numbers starting at +num+,
2137  * incremented by +step+ (defaulted to +1+) on each call.
2138  *
2139  * The loop finishes when the value to be passed to the block is greater than
2140  * +limit+ (if +step+ is positive) or less than +limit+ (if +step+ is
2141  * negative), where <i>limit</i> is defaulted to infinity.
2142  *

2144  * +step+ and +limit+ (default infinity) can be omitted. In the
2145  * fixed position argument style, zero as a step
2146  * (i.e. num.step(limit, 0)) is not allowed for historical
2147  * compatibility reasons.

2150  * counter.
2151  *
2152  * If any of the arguments are floating point numbers, all are converted to floats, and the loop is executed the following expression:
2153  *
2154  * floor(n + n*epsilon)+ 1
2155  *
2156  * Where the +n+ is the following:
2157  *
2158  * n = (limit - num)/step
2159  *
2160  * Otherwise, the loop starts at +num+, uses either the less-than (<) or
2161  * greater-than (>) operator to compare the counter against +limit+, and
2162  * increments itself using the <code>+</code> operator.
2163  *
2164  * If no block is given, an Enumerator is returned instead.
2165  *
2166  * For example:
2167  *
2168  * p 1.step.take(4)
2169  * p 10.step(by: -1).take(4)
2170  * 3.step(to: 5) { |i| print i, " " }
