215 /* Add to the loaded-features index all the required entries for
216  `feature`, located at `offset` in $LOADED_FEATURES. We add an
217  index entry at each string `short_feature` for which
218  feature == "#{prefix}#{short_feature}#{e}"
219  where `e` is empty or matches %r{^\.[^./]*$}, and `prefix` is empty
220  or ends in '/'. This maintains the invariant that `rb_feature_p()`
221  relies on for its fast lookup.
222 */
223 static void
224 features_index_add(VALUE feature, VALUE offset)
225 {
226  VALUE short_feature;
227  const char *feature_str, *feature_end, *ext, *p;
228 
229  feature_str = StringValuePtr(feature);
230  feature_end = feature_str + RSTRING_LEN(feature);
231 
232  for (ext = feature_end; ext > feature_str; ext--)
233  if (*ext == '.' || *ext == '/')
234  break;
235  if (*ext != '.')
236  ext = NULL;
237  /* Now `ext` points to the only string matching %r{^\.[^./]*$} that is
238  at the end of `feature`, or is NULL if there is no such string. */
239 
240  p = ext ? ext : feature_end;
241  while (1) {
242  p--;
243  while (p >= feature_str && *p != '/')
244  p--;
245  if (p < feature_str)
246  break;
247  /* Now *p == '/'. We reach this point for every '/' in `feature`. */
248  short_feature = rb_str_subseq(feature, p + 1 - feature_str, feature_end - p - 1);
249  features_index_add_single(short_feature, offset);
250  if (ext) {
251  short_feature = rb_str_subseq(feature, p + 1 - feature_str, ext - p - 1);
252  features_index_add_single(short_feature, offset);
253  }
254  }
255  features_index_add_single(feature, offset);
256  if (ext) {
257  short_feature = rb_str_subseq(feature, 0, ext - feature_str);
258  features_index_add_single(short_feature, offset);
259  }

300 /* This searches `load_path` for a value such that
301  name == "#{load_path[i]}/#{feature}"
302  if `feature` is a suffix of `name`, or otherwise
303  name == "#{load_path[i]}/#{feature}#{ext}"
304  for an acceptable string `ext`. It returns
305  `load_path[i].to_str` if found, else 0.
306 
307  If type is 's', then `ext` is acceptable only if IS_DLEXT(ext);
308  if 'r', then only if IS_RBEXT(ext); otherwise `ext` may be absent
309  or have any value matching `%r{^\.[^./]*$}`.
310 */
311 static VALUE
312 loaded_feature_path(const char *name, long vlen, const char *feature, long len,
313  int type, VALUE load_path)
314 {
315  long i;
316  long plen;
317  const char *e;
318 
319  if (vlen < len+1) return 0;
320  if (strchr(feature, '.') && !strncmp(name+(vlen-len), feature, len)) {
321  plen = vlen - len;
322  }
323  else {
324  for (e = name + vlen; name != e && *e != '.' && *e != '/'; --e);
325  if (*e != '.' ||
326  e-name < len ||
327  strncmp(e-len, feature, len))
328  return 0;
329  plen = e - name - len;
330  }
331  if (plen > 0 && name[plen-1] != '/') {
332  return 0;
333  }
334  if (type == 's' ? !IS_DLEXT(&name[plen+len]) :
335  type == 'r' ? !IS_RBEXT(&name[plen+len]) :
336  0) {
337  return 0;
338  }
339  /* Now name == "#{prefix}/#{feature}#{ext}" where ext is acceptable
340  (possibly empty) and prefix is some string of length plen. */
341 
342  if (plen > 0) --plen; /* exclude '.' */
343  for (i = 0; i < RARRAY_LEN(load_path); ++i) {
344  VALUE p = RARRAY_AREF(load_path, i);
345  const char *s = StringValuePtr(p);
346  long n = RSTRING_LEN(p);
347 
348  if (n != plen) continue;
349  if (n && strncmp(name, s, n)) continue;
350  return p;
351  }
352  return 0;
353 }

402  for some j, or

404  Here `e` is an "allowed" extension -- either empty or one
405  of the extensions accepted by IS_RBEXT, IS_SOEXT, or
406  IS_DLEXT. Further, if `ext && rb` then `IS_RBEXT(e)`,
407  and if `ext && !rb` then `IS_SOEXT(e) || IS_DLEXT(e)`.

412  preceded by an entry of the form
413  "#{features[i2]}" == "#{load_path[j2]}/#{feature}#{e2}"
414  where `e2` matches %r{^\.[^./]*$} but is not an allowed extension.
415  After a "distractor" entry of this form, only entries of the
416  form "#{feature}#{e}" are accepted.
417 
418  In `rb_provide_feature()` and `get_loaded_features_index()` we
419  maintain an invariant that the array `this_feature_index` will
420  point to every entry in `features` which has the form
421  "#{prefix}#{feature}#{e}"
422  where `e` is empty or matches %r{^\.[^./]*$}, and `prefix` is empty
423  or ends in '/'. This includes both match forms above, as well
424  as any distractors, so we may ignore all other entries in `features`.

491  };
492

511  }

514  }
