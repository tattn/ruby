848 static int
849 include_modules_at(const VALUE klass, VALUE c, VALUE module, int search_super)
850 {
851  VALUE p, iclass;
852  int method_changed = 0, constant_changed = 0;
853  const st_table *const klass_m_tbl = RCLASS_M_TBL(RCLASS_ORIGIN(klass));
854 
855  while (module) {
856  int superclass_seen = FALSE;
857 
858  if (RCLASS_ORIGIN(module) != module)
859  goto skip;
860  if (klass_m_tbl && klass_m_tbl == RCLASS_M_TBL(module))
861  return -1;
862  /* ignore if the module included already in superclasses */
863  for (p = RCLASS_SUPER(klass); p; p = RCLASS_SUPER(p)) {
864  int type = BUILTIN_TYPE(p);
865  if (type == T_ICLASS) {
866  if (RCLASS_M_TBL(p) == RCLASS_M_TBL(module)) {
867  if (!superclass_seen) {
868  c = p; /* move insertion point */
869  }
870  goto skip;
871  }
872  }
873  else if (type == T_CLASS) {
874  if (!search_super) break;
875  superclass_seen = TRUE;
876  }
877  }
878  iclass = rb_include_class_new(module, RCLASS_SUPER(c));
879  c = RCLASS_SET_SUPER(c, iclass);
880 
881  {
882  VALUE m = module;
883  if (BUILTIN_TYPE(m) == T_ICLASS) m = RBASIC(m)->klass;
884  rb_module_add_to_subclasses_list(m, iclass);
885  }
886 
887  if (FL_TEST(klass, RMODULE_IS_REFINEMENT)) {
888  VALUE refined_class =
889  rb_refinement_module_get_refined_class(klass);
890 
891  st_foreach(RMODULE_M_TBL(module), add_refined_method_entry_i,
892  (st_data_t) refined_class);
893  FL_SET(c, RMODULE_INCLUDED_INTO_REFINEMENT);
894  }
895  if (RMODULE_M_TBL(module) && RMODULE_M_TBL(module)->num_entries)
896  method_changed = 1;
897  if (RMODULE_CONST_TBL(module) && RMODULE_CONST_TBL(module)->num_entries)
898  constant_changed = 1;
899  skip:
900  module = RCLASS_SUPER(module);
901  }
902 
903  if (method_changed) rb_clear_method_cache_by_class(klass);
904  if (constant_changed) rb_clear_constant_cache();
905 
906  return method_changed;
907 }
