334 make_compile_option(rb_compile_option_t *option, VALUE opt)
335 {
336  if (opt == Qnil) {
337  *option = COMPILE_OPTION_DEFAULT;
338  }
339  else if (opt == Qfalse) {
340  *option = COMPILE_OPTION_FALSE;
341  }
342  else if (opt == Qtrue) {
343  int i;
344  for (i = 0; i < (int)(sizeof(rb_compile_option_t) / sizeof(int)); ++i)
345  ((int *)option)[i] = 1;
346  }
347  else if (CLASS_OF(opt) == rb_cHash) {
348  *option = COMPILE_OPTION_DEFAULT;
349 
350 #define SET_COMPILE_OPTION(o, h, mem) \
351  { VALUE flag = rb_hash_aref((h), ID2SYM(rb_intern(#mem))); \

353  else if (flag == Qfalse) { (o)->mem = 0; } \
354  }
355 #define SET_COMPILE_OPTION_NUM(o, h, mem) \
356  { VALUE num = rb_hash_aref(opt, ID2SYM(rb_intern(#mem))); \
357  if (!NIL_P(num)) (o)->mem = NUM2INT(num); \
358  }
359  SET_COMPILE_OPTION(option, opt, inline_const_cache);
360  SET_COMPILE_OPTION(option, opt, peephole_optimization);
361  SET_COMPILE_OPTION(option, opt, tailcall_optimization);
362  SET_COMPILE_OPTION(option, opt, specialized_instruction);
363  SET_COMPILE_OPTION(option, opt, operands_unification);

365  SET_COMPILE_OPTION(option, opt, stack_caching);
366  SET_COMPILE_OPTION(option, opt, trace_instruction);
367  SET_COMPILE_OPTION_NUM(option, opt, debug_level);
368 #undef SET_COMPILE_OPTION
369 #undef SET_COMPILE_OPTION_NUM
370  }

966 /*
967  * call-seq:
968  * iseq.to_a -> ary
969  *
970  * Returns an Array with 14 elements representing the instruction sequence
971  * with the following data:
972  *
973  * [magic]
974  * A string identifying the data format. <b>Always
975  * +YARVInstructionSequence/SimpleDataFormat+.</b>
976  *
977  * [major_version]
978  * The major version of the instruction sequence.
979  *
980  * [minor_version]
981  * The minor version of the instruction sequence.
982  *
983  * [format_type]
984  * A number identifying the data format. <b>Always 1</b>.
985  *
986  * [misc]
987  * A hash containing:
988  *
989  * [+:arg_size+]
990  * the total number of arguments taken by the method or the block (0 if
991  * _iseq_ doesn't represent a method or block)
992  * [+:local_size+]
993  * the number of local variables + 1
994  * [+:stack_max+]
995  * used in calculating the stack depth at which a SystemStackError is
996  * thrown.
997  *
998  * [#label]
999  * The name of the context (block, method, class, module, etc.) that this
1000  * instruction sequence belongs to.
1001  *
1002  * <code><main></code> if it's at the top level, <code><compiled></code> if
1003  * it was evaluated from a string.
1004  *
1005  * [#path]
1006  * The relative path to the Ruby file where the instruction sequence was
1007  * loaded from.
1008  *
1009  * <code><compiled></code> if the iseq was evaluated from a string.
1010  *
1011  * [#absolute_path]
1012  * The absolute path to the Ruby file where the instruction sequence was
1013  * loaded from.
1014  *
1015  * +nil+ if the iseq was evaluated from a string.
1016  *
1017  * [#first_lineno]
1018  * The number of the first source line where the instruction sequence was
1019  * loaded from.
1020  *
1021  * [type]
1022  * The type of the instruction sequence.
1023  *
1024  * Valid values are +:top+, +:method+, +:block+, +:class+, +:rescue+,
1025  * +:ensure+, +:eval+, +:main+, and +:defined_guard+.
1026  *
1027  * [locals]
1028  * An array containing the names of all arguments and local variables as
1029  * symbols.
1030  *
1031  * [params]
1032  * An Hash object containing parameter information.
1033  *
1034  * More info about these values can be found in +vm_core.h+.
1035  *
1036  * [catch_table]
1037  * A list of exceptions and control flow operators (rescue, next, redo,
1038  * break, etc.).
1039  *
1040  * [bytecode]
1041  * An array of arrays containing the instruction names and operands that
1042  * make up the body of the instruction sequence.
1043  *
1044  * Note that this format is MRI specific and version dependent.
1045  *
1046  */

1133 {

1135  char type = types[op_no];
1136  VALUE ret;
1137 
1138  switch (type) {
1139  case TS_OFFSET: /* LONG */
1140  ret = rb_sprintf("%"PRIdVALUE, (VALUE)(pos + len + op));
1141  break;
1142 
1143  case TS_NUM: /* ULONG */
1144  ret = rb_sprintf("%"PRIuVALUE, op);
1145  break;
1146 
1147  case TS_LINDEX:{
1148  if (insn == BIN(getlocal) || insn == BIN(setlocal)) {
1149  if (pnop) {
1150  const rb_iseq_t *diseq = iseq;
1151  VALUE level = *pnop, i;
1152 
1153  for (i = 0; i < level; i++) {
1154  diseq = diseq->parent_iseq;
1155  }
1156  ret = id_to_name(diseq->local_table[diseq->local_size - op], INT2FIX('*'));
1157  }
1158  else {
1159  ret = rb_sprintf("%"PRIuVALUE, op);
1160  }
1161  }
1162  else {
1163  ret = rb_inspect(INT2FIX(op));
1164  }
1165  break;
1166  }
1167  case TS_ID: /* ID (symbol) */
1168  op = ID2SYM(op);
1169 
1170  case TS_VALUE: /* VALUE */
1171  op = obj_resurrect(op);
1172  ret = rb_inspect(op);
1173  if (CLASS_OF(op) == rb_cISeq) {
1174  if (child) {
1175  rb_ary_push(child, op);
1176  }
1177  }
1178  break;
1179 
1180  case TS_ISEQ: /* iseq */
1181  {
1182  rb_iseq_t *iseq = (rb_iseq_t *)op;
1183  if (iseq) {
1184  ret = iseq->location.label;
1185  if (child) {
1186  rb_ary_push(child, iseq->self);
1187  }
1188  }
1189  else {
1190  ret = rb_str_new2("nil");
1191  }
1192  break;
1193  }
1194  case TS_GENTRY:
1195  {
1196  struct rb_global_entry *entry = (struct rb_global_entry *)op;
1197  ret = rb_str_dup(rb_id2str(entry->id));
1198  }
1199  break;
1200 
1201  case TS_IC:
1202  ret = rb_sprintf("<is:%"PRIdPTRDIFF">", (union iseq_inline_storage_entry *)op - iseq->is_entries);
1203  break;
1204 
1205  case TS_CALLINFO:
1206  {
1207  rb_call_info_t *ci = (rb_call_info_t *)op;
1208  VALUE ary = rb_ary_new();
1209 
1210  if (ci->mid) {
1211  rb_ary_push(ary, rb_sprintf("mid:%"PRIsVALUE, rb_id2str(ci->mid)));
1212  }
1213 
1214  rb_ary_push(ary, rb_sprintf("argc:%d", ci->orig_argc));
1215 
1216  if (ci->kw_arg) {
1217  rb_ary_push(ary, rb_sprintf("kw:%d", ci->kw_arg->keyword_len));
1218  }
1219  if (ci->blockiseq) {
1220  if (child) {
1221  rb_ary_push(child, ci->blockiseq->self);
1222  }
1223  rb_ary_push(ary, rb_sprintf("block:%"PRIsVALUE, ci->blockiseq->location.label));
1224  }
1225 
1226  if (ci->flag) {
1227  VALUE flags = rb_ary_new();
1228  if (ci->flag & VM_CALL_ARGS_SPLAT) rb_ary_push(flags, rb_str_new2("ARGS_SPLAT"));
1229  if (ci->flag & VM_CALL_ARGS_BLOCKARG) rb_ary_push(flags, rb_str_new2("ARGS_BLOCKARG"));
1230  if (ci->flag & VM_CALL_FCALL) rb_ary_push(flags, rb_str_new2("FCALL"));
1231  if (ci->flag & VM_CALL_VCALL) rb_ary_push(flags, rb_str_new2("VCALL"));
1232  if (ci->flag & VM_CALL_TAILCALL) rb_ary_push(flags, rb_str_new2("TAILCALL"));
1233  if (ci->flag & VM_CALL_SUPER) rb_ary_push(flags, rb_str_new2("SUPER"));
1234  if (ci->flag & VM_CALL_OPT_SEND) rb_ary_push(flags, rb_str_new2("SNED")); /* maybe not reachable */
1235  if (ci->flag & VM_CALL_ARGS_SIMPLE) rb_ary_push(flags, rb_str_new2("ARGS_SIMPLE")); /* maybe not reachable */
1236  rb_ary_push(ary, rb_ary_join(flags, rb_str_new2("|")));
1237  }
1238  ret = rb_sprintf("<callinfo!%"PRIsVALUE">", rb_ary_join(ary, rb_str_new2(", ")));
1239  }
1240  break;
1241 
1242  case TS_CDHASH:
1243  ret = rb_str_new2("<cdhash>");
1244  break;
1245 
1246  case TS_FUNCPTR:
1247  {
1248 #ifdef HAVE_DLADDR
1249  Dl_info info;
