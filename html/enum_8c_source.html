632  * call-seq:
633  * enum.inject(initial, sym) -> obj
634  * enum.inject(sym) -> obj
635  * enum.inject(initial) { |memo, obj| block } -> obj
636  * enum.inject { |memo, obj| block } -> obj
637  * enum.reduce(initial, sym) -> obj
638  * enum.reduce(sym) -> obj
639  * enum.reduce(initial) { |memo, obj| block } -> obj
640  * enum.reduce { |memo, obj| block } -> obj
641  *
642  * Combines all elements of <i>enum</i> by applying a binary
643  * operation, specified by a block or a symbol that names a
644  * method or operator.
645  *
646  * If you specify a block, then for each element in <i>enum</i>
647  * the block is passed an accumulator value (<i>memo</i>) and the element.
648  * If you specify a symbol instead, then each element in the collection
649  * will be passed to the named method of <i>memo</i>.
650  * In either case, the result becomes the new value for <i>memo</i>.
651  * At the end of the iteration, the final value of <i>memo</i> is the
652  * return value for the method.
653  *
654  * If you do not explicitly specify an <i>initial</i> value for <i>memo</i>,
655  * then the first element of collection is used as the initial value
656  * of <i>memo</i>.
657  *
658  *
659  * # Sum some numbers
660  * (5..10).reduce(:+) #=> 45
661  * # Same using a block and inject
662  * (5..10).inject { |sum, n| sum + n } #=> 45
663  * # Multiply some numbers
664  * (5..10).reduce(1, :*) #=> 151200
665  * # Same using a block
666  * (5..10).inject(1) { |product, n| product * n } #=> 151200
667  * # find the longest word
668  * longest = %w{ cat sheep bear }.inject do |memo, word|
669  * memo.length > word.length ? memo : word
670  * end
671  * longest #=> "sheep"
672  *
673  */

925 /*
926  * call-seq:
927  * enum.sort_by { |obj| block } -> array
928  * enum.sort_by -> an_enumerator
929  *
930  * Sorts <i>enum</i> using a set of keys generated by mapping the
931  * values in <i>enum</i> through the given block.
932  *
933  * If no block is given, an enumerator is returned instead.
934  *
935  * %w{apple pear fig}.sort_by { |word| word.length}
936  * #=> ["fig", "pear", "apple"]
937  *
938  * The current implementation of <code>sort_by</code> generates an
939  * array of tuples containing the original collection element and the
940  * mapped value. This makes <code>sort_by</code> fairly expensive when
941  * the keysets are simple.
942  *
943  * require 'benchmark'
944  *
945  * a = (1..100000).map { rand(100000) }
946  *
947  * Benchmark.bm(10) do |b|
948  * b.report("Sort") { a.sort }
949  * b.report("Sort by") { a.sort_by { |a| a } }
950  * end
951  *
952  * <em>produces:</em>
953  *
954  * user system total real
955  * Sort 0.180000 0.000000 0.180000 ( 0.175469)
956  * Sort by 1.980000 0.040000 2.020000 ( 2.013586)
957  *
958  * However, consider the case where comparing the keys is a non-trivial
959  * operation. The following code sorts some files on modification time
960  * using the basic <code>sort</code> method.
961  *
962  * files = Dir["*"]
963  * sorted = files.sort { |a, b| File.new(a).mtime <=> File.new(b).mtime }
964  * sorted #=> ["mon", "tues", "wed", "thurs"]
965  *
966  * This sort is inefficient: it generates two new <code>File</code>
967  * objects during every comparison. A slightly better technique is to
968  * use the <code>Kernel#test</code> method to generate the modification
969  * times directly.
970  *
971  * files = Dir["*"]
972  * sorted = files.sort { |a, b|
973  * test(?M, a) <=> test(?M, b)
974  * }
975  * sorted #=> ["mon", "tues", "wed", "thurs"]
976  *
977  * This still generates many unnecessary <code>Time</code> objects. A
978  * more efficient technique is to cache the sort keys (modification
979  * times in this case) before the sort. Perl users often call this
980  * approach a Schwartzian Transform, after Randal Schwartz. We
981  * construct a temporary array, where each element is an array
982  * containing our sort key along with the filename. We sort this array,
983  * and then extract the filename from the result.
984  *
985  * sorted = Dir["*"].collect { |f|
986  * [test(?M, f), f]
987  * }.sort.collect { |f| f[1] }
988  * sorted #=> ["mon", "tues", "wed", "thurs"]
989  *
990  * This is exactly what <code>sort_by</code> does internally.
991  *
992  * sorted = Dir["*"].sort_by { |f| test(?M, f) }
993  * sorted #=> ["mon", "tues", "wed", "thurs"]
994  */

1186 nmin_filter(struct nmin_data *data)
1187 {
1188  long n;
1189  VALUE *beg;
1190  int eltsize;
1191  long numelts;
1192 
1193  long left, right;
1194 
1195  long i, j;
1196 
1197  if (data->curlen <= data->n)
1198  return;
1199 
1200  n = data->n;
1201  beg = RARRAY_PTR(data->buf);
1202  eltsize = data->by ? 2 : 1;
1203  numelts = data->curlen;
1204 
1205  left = 0;
1206  right = numelts-1;
1207 
1208 #define GETPTR(i) (beg+(i)*eltsize)
1209 
1210 #define SWAP(i, j) do { \
1211  VALUE tmp[2]; \
1212  memcpy(tmp, GETPTR(i), sizeof(VALUE)*eltsize); \
1213  memcpy(GETPTR(i), GETPTR(j), sizeof(VALUE)*eltsize); \
1214  memcpy(GETPTR(j), tmp, sizeof(VALUE)*eltsize); \
1215 } while (0)
1216 
1217  while (1) {
1218  long pivot_index = left + (right-left)/2;
1219  long store_index;
1220  long num_pivots = 1;
1221 
1222  SWAP(pivot_index, right);
1223  pivot_index = right;
1224 
1225  store_index = left;
1226  i = left;
1227  while (i <= right-num_pivots) {
1228  int c = data->cmpfunc(GETPTR(i), GETPTR(pivot_index), data);
1229  if (data->rev)
1230  c = -c;
1231  if (c == 0) {
1232  SWAP(i, right-num_pivots);
1233  num_pivots++;
1234  continue;
1235  }
1236  if (c < 0) {
1237  SWAP(i, store_index);
1238  store_index++;
1239  }
1240  i++;
1241  }
1242  j = store_index;
1243  for (i = right; right-num_pivots < i; i--) {
1244  if (i <= j)
1245  break;
1246  SWAP(j, i);
1247  j++;
1248  }
1249 
1250  if (store_index <= n && n <= store_index+num_pivots)
1251  break;
1252 
1253  if (n < store_index) {
1254  right = store_index-1;
1255  }
1256  else {
1257  left = store_index+num_pivots;
1258  }
1259  }
1260 #undef GETPTR
1261 #undef SWAP
1262 
1263  data->curlen = data->n;
1264  rb_ary_resize(data->buf, data->n * eltsize);
1265  data->limit = RARRAY_PTR(data->buf)[(data->n-1)*eltsize];
1266 }
