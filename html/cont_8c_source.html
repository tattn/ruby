20  * This hack make Fiber context switch faster (x2 or more).
21  * However, it decrease maximum number of Fiber. For example, on the
22  * 32bit POSIX OS, ten or twenty thousands Fiber can be created.
23  *
24  * Details is reported in the paper "A Fast Fiber Implementation for Ruby 1.9"
25  * in Proc. of 51th Programming Symposium, pp.21--28 (2010) (in Japanese).
26  */

198  /* cont */

369 cont_save_machine_stack(rb_thread_t *th, rb_context_t *cont)
370 {
371  size_t size;
372 
373  SET_MACHINE_STACK_END(&th->machine.stack_end);

482  rb_context_t *cont;

506  {

508  int size = 0;
509  rb_ensure_entry_t *entry;

517  }

519  }
520

522  volatile VALUE value;
523 
524  VAR_INITIALIZED(cont);
525  value = cont->value;
526  if (cont->argc == -1) rb_exc_raise(value);
527  cont->value = Qnil;
528  *stat = 1;
529  return value;
530  }

532  *stat = 0;

618 static char*
619 fiber_machine_stack_alloc(size_t size)
620 {
621  char *ptr;
622 
623  if (machine_stack_cache_index > 0) {
624  if (machine_stack_cache[machine_stack_cache_index - 1].size == (size / sizeof(VALUE))) {
625  ptr = machine_stack_cache[machine_stack_cache_index - 1].ptr;
626  machine_stack_cache_index--;
627  machine_stack_cache[machine_stack_cache_index].ptr = NULL;
628  machine_stack_cache[machine_stack_cache_index].size = 0;
629  }
630  else{
631  /* TODO handle multiple machine stack size */
632  rb_bug("machine_stack_cache size is not canonicalized");
633  }
634  }
635  else {
636  void *page;
637  STACK_GROW_DIR_DETECTION;
638 
639  errno = 0;
640  ptr = mmap(NULL, size, PROT_READ | PROT_WRITE, FIBER_STACK_FLAGS, -1, 0);
641  if (ptr == MAP_FAILED) {
642  rb_raise(rb_eFiberError, "can't alloc machine stack to fiber: %s", strerror(errno));
643  }
644 
645  /* guard page setup */

866 /*
867  * Document-class: Continuation
868  *
869  * Continuation objects are generated by Kernel#callcc,
870  * after having +require+d <i>continuation</i>. They hold
871  * a return address and execution context, allowing a nonlocal return
872  * to the end of the <code>callcc</code> block from anywhere within a
873  * program. Continuations are somewhat analogous to a structured
874  * version of C's <code>setjmp/longjmp</code> (although they contain
875  * more state, so you might consider them closer to threads).
876  *
877  * For instance:
878  *
879  * require "continuation"
880  * arr = [ "Freddie", "Herbie", "Ron", "Max", "Ringo" ]
881  * callcc{|cc| $cc = cc}
882  * puts(message = arr.shift)
883  * $cc.call unless message =~ /Max/
884  *
885  * <em>produces:</em>
886  *
887  * Freddie
888  * Herbie
889  * Ron
890  * Max
891  *
892  * This (somewhat contrived) example allows the inner loop to abandon
893  * processing early:
894  *
895  * require "continuation"
896  * callcc {|cont|
897  * for i in 0..4
898  * print "\n#{i}: "
899  * for j in i*5...(i+1)*5
900  * cont.call() if j == 17
901  * printf "%3d", j
902  * end
903  * end
904  * }
905  * puts
906  *
907  * <em>produces:</em>
908  *
909  * 0: 0 1 2 3 4
910  * 1: 5 6 7 8 9
911  * 2: 10 11 12 13 14
912  * 3: 15 16
913  */
914 
915 /*
916  * call-seq:
917  * callcc {|cont| block } -> obj
918  *
919  * Generates a Continuation object, which it passes to
920  * the associated block. You need to <code>require
921  * 'continuation'</code> before using this method. Performing a
922  * <em>cont</em><code>.call</code> will cause the #callcc
923  * to return (as will falling through the end of the block). The
924  * value returned by the #callcc is the value of the
925  * block, or the value passed to <em>cont</em><code>.call</code>. See
926  * class Continuation for more details. Also see
927  * Kernel#throw for an alternative mechanism for
928  * unwinding a call stack.
929  */

936 
937  if (called) {
938  return val;
939  }
940  else {
941  return rb_yield(val);
942  }
943 }

981 static inline void
982 rollback_ensure_stack(VALUE self,rb_ensure_list_t *current,rb_ensure_entry_t *target)
983 {
984  rb_ensure_list_t *p;
985  rb_ensure_entry_t *entry;
986  size_t i;
987  size_t cur_size;
988  size_t target_size;
989  size_t base_point;
990  VALUE (*func)(ANYARGS);
991 
992  cur_size = 0;
993  for (p=current; p; p=p->next)
994  cur_size++;
995  target_size = 0;
996  for (entry=target; entry->marker; entry++)
997  target_size++;
998 
999  /* search common stack point */
1000  p = current;
1001  base_point = cur_size;
1002  while (base_point) {
1003  if (target_size >= base_point &&
1004  p->entry.marker == target[target_size - base_point].marker)
1005  break;
1006  base_point --;
1007  p = p->next;
1008  }
1009 
1010  /* rollback function check */
1011  for (i=0; i < target_size - base_point; i++) {
1012  if (!lookup_rollback_func(target[i].e_proc)) {
1013  rb_raise(rb_eRuntimeError, "continuation called from out of critical rb_ensure scope");
1014  }
1015  }
1016  /* pop ensure stack */
1017  while (cur_size > base_point) {
1018  /* escape from ensure block */
1019  (*current->entry.e_proc)(current->entry.data2);
1020  current = current->next;
1021  cur_size--;
1022  }
1023  /* push ensure stack */
1024  while (i--) {
1025  func = (VALUE (*)(ANYARGS)) lookup_rollback_func(target[i].e_proc);
1026  if ((VALUE)func != Qundef) {
1027  (*func)(target[i].data2);
1028  }
1029  }
1030 }

1082 /*
1083  * Document-class: Fiber
1084  *
1085  * Fibers are primitives for implementing light weight cooperative
1086  * concurrency in Ruby. Basically they are a means of creating code blocks
1087  * that can be paused and resumed, much like threads. The main difference
1088  * is that they are never preempted and that the scheduling must be done by
1089  * the programmer and not the VM.
1090  *
1091  * As opposed to other stackless light weight concurrency models, each fiber
1092  * comes with a small 4KB stack. This enables the fiber to be paused from deeply
1093  * nested function calls within the fiber block.
1094  *
1095  * When a fiber is created it will not run automatically. Rather it must be
1096  * be explicitly asked to run using the <code>Fiber#resume</code> method.
1097  * The code running inside the fiber can give up control by calling
1098  * <code>Fiber.yield</code> in which case it yields control back to caller
1099  * (the caller of the <code>Fiber#resume</code>).
1100  *
1101  * Upon yielding or termination the Fiber returns the value of the last
1102  * executed expression
1103  *
1104  * For instance:
1105  *
1106  * fiber = Fiber.new do
1107  * Fiber.yield 1
1108  * 2
1109  * end
1110  *
1111  * puts fiber.resume
1112  * puts fiber.resume
1113  * puts fiber.resume
1114  *
1115  * <em>produces</em>
1116  *
1117  * 1
1118  * 2
1119  * FiberError: dead fiber called
1120  *
1121  * The <code>Fiber#resume</code> method accepts an arbitrary number of
1122  * parameters, if it is the first call to <code>resume</code> then they
1123  * will be passed as block arguments. Otherwise they will be the return
1124  * value of the call to <code>Fiber.yield</code>
1125  *
1126  * Example:
1127  *
1128  * fiber = Fiber.new do |first|
1129  * second = Fiber.yield first + 2
1130  * end
1131  *
1132  * puts fiber.resume 10
1133  * puts fiber.resume 14
1134  * puts fiber.resume 18
1135  *
1136  * <em>produces</em>
1137  *
1138  * 12
1139  * 14
1140  * FiberError: dead fiber called
1141  *
1142  */

1399 {

1404  if (th->fiber == fib) {

1406  * because destination fiber is same as current fiber
1407  */

1409  }
1410

1413  }
1414  else if (cont->saved_thread.protect_tag != th->protect_tag) {
1415  rb_raise(rb_eFiberError, "fiber called across stack rewinding barrier");

1421

1424  /* and the terminated fiber's return_fiber() is already dead) */

1426  cont->argc = -1;
1427  cont->value = value;
1428 #if FIBER_USE_NATIVE
1429  fiber_setcontext(th->root_fiber, th->fiber);
1430 #else
1431  cont_restore_0(cont, &value);
1432 #endif

1434  }
1435

1437  fib->prev = fiber_current();
1438  }

1441  th->trace_arg = cont->saved_thread.trace_arg;
1442  }
1443 
1444  cont->argc = argc;
1445  cont->value = make_passing_arg(argc, argv);
1446
