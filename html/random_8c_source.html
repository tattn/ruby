113 /* initializes state[N] with a seed */
114 static void
115 init_genrand(struct MT *mt, unsigned int s)
116 {
117  int j;
118  mt->state[0] = s & 0xffffffffU;
119  for (j=1; j<N; j++) {
120  mt->state[j] = (1812433253U * (mt->state[j-1] ^ (mt->state[j-1] >> 30)) + j);
121  /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
122  /* In the previous versions, MSBs of the seed affect */
123  /* only MSBs of the array state[]. */
124  /* 2002/01/09 modified by Makoto Matsumoto */
125  mt->state[j] &= 0xffffffff; /* for >32 bit machines */
126  }
127  mt->left = 1;
128  mt->next = mt->state + N;
129 }

135 static void
136 init_by_array(struct MT *mt, unsigned int init_key[], int key_length)
137 {
138  int i, j, k;
139  init_genrand(mt, 19650218U);
140  i=1; j=0;
141  k = (N>key_length ? N : key_length);
142  for (; k; k--) {
143  mt->state[i] = (mt->state[i] ^ ((mt->state[i-1] ^ (mt->state[i-1] >> 30)) * 1664525U))
144  + init_key[j] + j; /* non linear */
145  mt->state[i] &= 0xffffffffU; /* for WORDSIZE > 32 machines */
146  i++; j++;
147  if (i>=N) { mt->state[0] = mt->state[N-1]; i=1; }
148  if (j>=key_length) j=0;
149  }
150  for (k=N-1; k; k--) {
151  mt->state[i] = (mt->state[i] ^ ((mt->state[i-1] ^ (mt->state[i-1] >> 30)) * 1566083941U))
152  - i; /* non linear */
153  mt->state[i] &= 0xffffffffU; /* for WORDSIZE > 32 machines */
154  i++;
155  if (i>=N) { mt->state[0] = mt->state[N-1]; i=1; }
156  }
157 
158  mt->state[0] = 0x80000000U; /* MSB is 1; assuring non-zero initial array */
159 }

161 static void
162 next_state(struct MT *mt)
163 {
164  unsigned int *p = mt->state;
165  int j;
166 
167  mt->left = N;
168  mt->next = mt->state;
169 
170  for (j=N-M+1; --j; p++)
171  *p = p[M] ^ TWIST(p[0], p[1]);
172 
173  for (j=M; --j; p++)
174  *p = p[M-N] ^ TWIST(p[0], p[1]);
175 
176  *p = p[M-N] ^ TWIST(p[0], mt->state[0]);
177 }

180 static unsigned int
181 genrand_int32(struct MT *mt)
182 {
183  /* mt must be initialized */
184  unsigned int y;
185 
186  if (--mt->left <= 0) next_state(mt);
187  y = *mt->next++;
188 
189  /* Tempering */
190  y ^= (y >> 11);
191  y ^= (y << 7) & 0x9d2c5680;
192  y ^= (y << 15) & 0xefc60000;
193  y ^= (y >> 18);

277 static double
278 int_pair_to_real_inclusive(uint32_t a, uint32_t b)
279 {
280  VALUE x;
281  VALUE m;
282  uint32_t xary[2], mary[2];
283  double r;
284 
285  /* (a << 32) | b */
286  xary[0] = a;
287  xary[1] = b;
288  x = rb_integer_unpack(xary, 2, sizeof(uint32_t), 0,
289  INTEGER_PACK_MSWORD_FIRST|INTEGER_PACK_NATIVE_BYTE_ORDER|
290  INTEGER_PACK_FORCE_BIGNUM);
291 
292  /* (1 << 53) | 1 */
293  mary[0] = 0x00200000;
294  mary[1] = 0x00000001;
295  m = rb_integer_unpack(mary, 2, sizeof(uint32_t), 0,
296  INTEGER_PACK_MSWORD_FIRST|INTEGER_PACK_NATIVE_BYTE_ORDER|
297  INTEGER_PACK_FORCE_BIGNUM);
298 
299  x = rb_big_mul(x, m);
300  if (FIXNUM_P(x)) {
301 #if CHAR_BIT * SIZEOF_LONG > 64
302  r = (double)(FIX2ULONG(x) >> 64);
303 #else
304  return 0.0;
305 #endif
306  }
307  else {
308  uint32_t uary[4];
309  rb_integer_pack(x, uary, numberof(uary), sizeof(uint32_t), 0,
310  INTEGER_PACK_MSWORD_FIRST|INTEGER_PACK_NATIVE_BYTE_ORDER);
311  /* r = x >> 64 */
312  r = (double)uary[0] * (0x10000 * (double)0x10000) + (double)uary[1];
313  }
314  return ldexp(r, -53);
315 }

380 static VALUE
381 rand_init(struct MT *mt, VALUE vseed)
382 {
383  volatile VALUE seed;
384  uint32_t buf0[SIZEOF_LONG / SIZEOF_INT32 * 4], *buf = buf0;
385  size_t len;
386  int sign;
387 
388  seed = rb_to_int(vseed);
389 
390  len = rb_absint_numwords(seed, 32, NULL);
391  if (len > numberof(buf0))
392  buf = ALLOC_N(unsigned int, len);
393  sign = rb_integer_pack(seed, buf, len, sizeof(uint32_t), 0,
394  INTEGER_PACK_LSWORD_FIRST|INTEGER_PACK_NATIVE_BYTE_ORDER);
395  if (sign < 0)
396  sign = -sign;
397  if (len == 0) {
398  buf[0] = 0;
399  len = 1;
400  }
401  if (len <= 1) {
402  init_genrand(mt, buf[0]);
403  }
404  else {
405  if (sign != 2 && buf[len-1] == 1) /* remove leading-zero-guard */
406  len--;
407  init_by_array(mt, buf, (int)len);
408  }
409  if (buf != buf0) xfree(buf);
410  return seed;
411 }
