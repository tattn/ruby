93 static int
94 fc_i(st_data_t k, st_data_t v, st_data_t a)
95 {
96  ID key = (ID)k;
97  rb_const_entry_t *ce = (rb_const_entry_t *)v;
98  struct fc_result *res = (struct fc_result *)a;
99  VALUE value = ce->value;
100  if (!rb_is_const_id(key)) return ST_CONTINUE;
101 
102  if (value == res->klass && (!res->preferred || key == res->preferred)) {
103  res->path = fc_path(res, key);
104  return ST_STOP;
105  }
106  if (RB_TYPE_P(value, T_MODULE) || RB_TYPE_P(value, T_CLASS)) {
107  if (!RCLASS_CONST_TBL(value)) return ST_CONTINUE;
108  else {
109  struct fc_result arg;
110  struct fc_result *list;
111 
112  list = res;
113  while (list) {
114  if (list->track == value) return ST_CONTINUE;
115  list = list->prev;
116  }
117 
118  arg.name = key;
119  arg.preferred = res->preferred;
120  arg.path = 0;
121  arg.klass = res->klass;
122  arg.track = value;
123  arg.prev = res;
124  st_foreach(RCLASS_CONST_TBL(value), fc_i, (st_data_t)&arg);
125  if (arg.path) {
126  res->path = arg.path;

128  }
129  }

176 classname(VALUE klass, int *permanent)
177 {
178  VALUE path = Qnil;
179  st_data_t n;
180 
181  if (!klass) klass = rb_cObject;
182  *permanent = 1;

184  if (!st_lookup(RCLASS_IV_TBL(klass), (st_data_t)classpath, &n)) {
185  ID cid = 0;

187  VALUE cname = (VALUE)n;

189  if (cid) path = find_class_path(klass, cid);
190  }
191  if (NIL_P(path)) {

195  if (!cid) {
196  return Qnil;
197  }

199  path = rb_id2str(cid);

201  }
202  *permanent = 0;

387 rb_path_to_class(VALUE pathname)
388 {
389  rb_encoding *enc = rb_enc_get(pathname);
390  const char *pbeg, *p, *path = RSTRING_PTR(pathname);
391  ID id;
392  VALUE c = rb_cObject;
393 
394  if (!rb_enc_asciicompat(enc)) {
395  rb_raise(rb_eArgError, "invalid class path encoding (non ASCII)");
396  }
397  pbeg = p = path;
398  if (path[0] == '#') {
399  rb_raise(rb_eArgError, "can't retrieve anonymous class %"PRIsVALUE,
400  QUOTE(pathname));
401  }
402  while (*p) {
403  while (*p && *p != ':') p++;
404  id = rb_check_id_cstr(pbeg, p-pbeg, enc);
405  if (p[0] == ':') {
406  if (p[1] != ':') goto undefined_class;
407  p += 2;
408  pbeg = p;
409  }
410  if (!id || !rb_const_defined_at(c, id)) {
411  undefined_class:
412  rb_raise(rb_eArgError, "undefined class/module % "PRIsVALUE,
413  rb_str_subseq(pathname, 0, p-path));
414  }
415  c = rb_const_get_at(c, id);
416  if (!RB_TYPE_P(c, T_MODULE) && !RB_TYPE_P(c, T_CLASS)) {
417  rb_raise(rb_eTypeError, "%"PRIsVALUE" does not refer to class/module",
418  pathname);
419  }
420  }
421  RB_GC_GUARD(pathname);
422 
423  return c;
424 }

681  * call-seq:
682  * trace_var(symbol, cmd ) -> nil
683  * trace_var(symbol) {|val| block } -> nil
684  *
685  * Controls tracing of assignments to global variables. The parameter
686  * +symbol+ identifies the variable (as either a string name or a
687  * symbol identifier). _cmd_ (which may be a string or a
688  * +Proc+ object) or block is executed whenever the variable
689  * is assigned. The block or +Proc+ object receives the
690  * variable's new value as a parameter. Also see
691  * <code>Kernel::untrace_var</code>.
692  *
693  * trace_var :$_, proc {|v| puts "$_ is now '#{v}'" }
694  * $_ = "hello"
695  * $_ = ' there'
696  *
697  * <em>produces:</em>
698  *
699  * $_ is now 'hello'

1793  * call-seq:
1794  * mod.const_missing(sym) -> obj
1795  *
1796  * Invoked when a reference is made to an undefined constant in
1797  * <i>mod</i>. It is passed a symbol for the undefined constant, and
1798  * returns a value to be used for that constant. The
1799  * following code is an example of the same:
1800  *
1801  * def Foo.const_missing(name)
1802  * name # return the constant name as Symbol
1803  * end

1805  * Foo::UNDEFINED_CONST #=> :UNDEFINED_CONST: symbol returned

1808  * it attempts to load a file whose name is the lowercase version of the

1812  * Module#autoload.
1813  *
1814  * def Object.const_missing(name)
1815  * @looked_for ||= {}

1817  * raise "Class not found: #{name}" if @looked_for[str_name]
1818  * @looked_for[str_name] = 1
1819  * file = str_name.downcase
1820  * require file
1821  * klass = const_get(name)
1822  * return klass if klass
1823  * raise "Class not found: #{name}"
1824  * end
1825  *
1826  */
1827
